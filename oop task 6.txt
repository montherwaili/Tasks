    CLASS Instrument
        PRIVATE ATTRIBUTES:
            brand : String
            price : Double

        CONSTRUCTOR(brnd, prc)
            SET this.brand TO brnd
            SET this.price TO prc
        ENDCONSTRUCTOR

        METHOD getBrand()
            RETURN this.brand
        ENDMETHOD

        METHOD getPrice()
            RETURN this.price
        ENDMETHOD

        ABSTRACT METHOD describePlayStyle()
    ENDCLASS

    CLASS Guitar EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfStrings : Integer

        CONSTRUCTOR(brnd, prc, strings)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfStrings TO strings
        ENDCONSTRUCTOR

        METHOD describePlayStyle()
            RETURN "strummed or plucked"
        ENDMETHOD
    ENDCLASS

    CLASS DrumKit EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfPieces : Integer

        CONSTRUCTOR(brnd, prc, pieces)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfPieces TO pieces
        ENDCONSTRUCTOR

        METHOD describePlayStyle()

        ENDMETHOD
    ENDCLASS

    CLASS Keyboard EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfKeys : Integer

        CONSTRUCTOR(brnd, prc, keys)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfKeys TO keys

        ENDMETHOD
    ENDCLASS

    CLASS MusicStore
        PRIVATE ATTRIBUTES:
            storeID : Integer
            storeName : String
            instrumentsStock : List of Instrument

        CONSTRUCTOR(id, name)
            SET this.storeID TO id
            SET this.storeName TO name
            SET this.instrumentsStock TO LIST []
        ENDCONSTRUCTOR

        METHOD addInstrument(instrumentObj)
            ADD instrumentObj TO this.instrumentsStock
            DISPLAY "Instrument successfully added to the store stock inventory."
        ENDMETHOD

        METHOD calculateTotalStockValue()
            SET totalValue TO 0.0
            SET totalInstruments TO SIZE OF this.instrumentsStock
            
            REPEAT FROM i = 0 TO totalInstruments - 1
                SET totalValue TO totalValue + CALL this.instrumentsStock[i].getPrice()
            ENDREPEAT
            
            RETURN totalValue
        ENDMETHOD

        METHOD getStockCount()
            RETURN SIZE OF this.instrumentsStock
        ENDMETHOD
    ENDCLASS

    CREATE FenderGuitar FROM Guitar("Fender", 450.00, 6)
    CREATE YamahaKeyboard FROM Keyboard("Yamaha", 720.00, 88)

    CALL MyMusicStore.addInstrument(FenderGuitar)
    CALL MyMusicStore.addInstrument(YamahaKeyboard)

    DISPLAY "Total Active Instruments in Stock: " + CALL MyMusicStore.getStockCount()
    DISPLAY "Total Inventory Financial Valuation: OMR " + CALL MyMusicStore.calculateTotalStockValue()

BEGIN

    CLASS Workout
        PRIVATE ATTRIBUTES:
            workoutDate : String
            durationMinutes : Integer

        CONSTRUCTOR(wDate, duration)
            SET this.workoutDate TO wDate
            SET this.durationMinutes TO duration
        ENDCONSTRUCTOR

        METHOD getDurationMinutes()
            RETURN this.durationMinutes
        ENDMETHOD

        ABSTRACT METHOD calculateCalories()
    ENDCLASS

    CLASS RunningWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            distanceKM : Double
            paceMinutesPerKM : Double

        CONSTRUCTOR(wDate, duration, dist, pc)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.distanceKM TO dist
            SET this.paceMinutesPerKM TO pc
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            SET duration TO CALL this.getDurationMinutes()
            RETURN (distanceKM * 60.0) + (duration * 0.5)
        ENDMETHOD
    ENDCLASS

    CLASS CyclingWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            distanceKM : Double
            resistanceLevel : Integer

        CONSTRUCTOR(wDate, duration, dist, resist)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.distanceKM TO dist
            SET this.resistanceLevel TO resist
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            SET duration TO CALL this.getDurationMinutes()
            RETURN (distanceKM * (35.0 + resistanceLevel)) + (duration * 0.4)
        ENDMETHOD
    ENDCLASS

    CLASS WeightTrainingWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            totalSets : Integer
            repsPerSet : Integer

        CONSTRUCTOR(wDate, duration, sets, reps)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.totalSets TO sets
            SET this.repsPerSet TO reps
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            SET duration TO CALL this.getDurationMinutes()
            RETURN (totalSets * repsPerSet * 0.4) + (duration * 3.0)
        ENDMETHOD
    ENDCLASS

    CLASS FitnessLog
        PRIVATE ATTRIBUTES:
            logID : Integer
            workoutsList : List of Workout

        CONSTRUCTOR(id)
            SET this.logID TO id
            SET this.workoutsList TO LIST []
        ENDCONSTRUCTOR

        METHOD logWorkout(workoutObj)
            ADD workoutObj TO this.workoutsList
            DISPLAY "Workout session successfully saved to fitness log."
        ENDMETHOD

        METHOD calculateTotalCaloriesBurned()
            SET totalBurned TO 0.0
            SET totalSessions TO SIZE OF this.workoutsList
            
            REPEAT FROM i = 0 TO totalSessions - 1
                SET totalBurned TO totalBurned + CALL this.workoutsList[i].calculateCalories()
            ENDREPEAT
            
            RETURN totalBurned
        ENDMETHOD
    ENDCLASS

    CREATE MyFitnessLog FROM FitnessLog(1)
    CREATE Session_1 FROM RunningWorkout("2026-07-12", 45, 8.5, 5.2)

    CALL MyFitnessLog.logWorkout(Session_1)
    DISPLAY "Total Cumulative Calories Burned: " + CALL MyFitnessLog.calculateTotalCaloriesBurned()

END

    CLASS MeterReading
        PRIVATE ATTRIBUTES:
            startReading : Integer
            endReading : Integer

        CONSTRUCTOR(start, end)
            SET this.startReading TO start
            SET this.endReading TO end
        ENDCONSTRUCTOR

        METHOD calculateConsumption()
            RETURN this.endReading - this.startReading
        ENDMETHOD
    ENDCLASS

    CLASS Bill
        PRIVATE ATTRIBUTES:
            billingPeriod : String
            dueDate : String
            meterReadingObj : MeterReading

        CONSTRUCTOR(period, due, reading)
            SET this.billingPeriod TO period
            SET this.dueDate TO due
            SET this.meterReadingObj TO reading
        ENDCONSTRUCTOR

        METHOD getMeterReading()
            RETURN this.meterReadingObj
        ENDMETHOD

        ABSTRACT METHOD calculateAmountOwed()
    ENDCLASS

    CLASS ElectricityBill EXTENDS Bill
        PRIVATE ATTRIBUTES:
            peakRate : Double
            offPeakRate : Double
            isPeakTime : Boolean

        CONSTRUCTOR(period, due, reading, pRate, opRate, peakFlag)
            CALL SUPER_CONSTRUCTOR(period, due, reading)
            SET this.peakRate TO pRate
            SET this.offPeakRate TO opRate
            SET this.isPeakTime TO peakFlag
        ENDCONSTRUCTOR

        METHOD calculateAmountOwed()
            SET reading TO CALL this.getMeterReading()
            SET consumption TO CALL reading.calculateConsumption()
            IF this.isPeakTime Equals TRUE THEN
                RETURN consumption * this.peakRate
            ELSE
                RETURN consumption * this.offPeakRate
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS WaterBill EXTENDS Bill
        PRIVATE ATTRIBUTES:
            flatRatePerLiter : Double
            fixedStandingCharge : Double

        CONSTRUCTOR(period, due, reading, rate, charge)
            CALL SUPER_CONSTRUCTOR(period, due, reading)
            SET this.flatRatePerLiter TO rate
            SET this.fixedStandingCharge TO charge
        ENDCONSTRUCTOR

        METHOD calculateAmountOwed()
            SET reading TO CALL this.getMeterReading()
            SET consumption TO CALL reading.calculateConsumption()
            RETURN (consumption * this.flatRatePerLiter) + this.fixedStandingCharge
        ENDMETHOD
    ENDCLASS

    CLASS GasBill EXTENDS Bill
        PRIVATE ATTRIBUTES:
            standardRate : Double
            isWinterSeason : Boolean

        CONSTRUCTOR(period, due, reading, rate, winterFlag)
            CALL SUPER_CONSTRUCTOR(period, due, reading)
            SET this.standardRate TO rate
            SET this.isWinterSeason TO winterFlag
        ENDCONSTRUCTOR

        METHOD calculateAmountOwed()
            SET reading TO CALL this.getMeterReading()
            SET consumption TO CALL reading.calculateConsumption()
            SET baselineAmount TO consumption * this.standardRate
            IF this.isWinterSeason Equals TRUE THEN
                RETURN baselineAmount + (baselineAmount * 0.15)
            ELSE
                RETURN baselineAmount
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS CustomerAccount
        PRIVATE ATTRIBUTES:
            accountID : Integer
            billsHistory : List of Bill

        CONSTRUCTOR(id)
            SET this.accountID TO id
            SET this.billsHistory TO LIST []
        ENDCONSTRUCTOR

        METHOD issueBill(billObj)
            ADD billObj TO this.billsHistory
            DISPLAY "Utility bill successfully issued and logged under customer account."
        ENDMETHOD

        METHOD calculateTotalOwedBalance()
            SET totalOwed TO 0.0
            SET totalBills TO SIZE OF this.billsHistory
            
            REPEAT FROM i = 0 TO totalBills - 1
                SET totalOwed TO totalOwed + CALL this.billsHistory[i].calculateAmountOwed()
            ENDREPEAT
            
            RETURN totalOwed
        ENDMETHOD
    ENDCLASS

    CREATE MontherAccount FROM CustomerAccount(7015)
    CREATE CurrentReading FROM MeterReading(12400, 12950)
    CREATE PowerInvoice FROM ElectricityBill("July 2026", "2026-08-01", CurrentReading, 0.024, 0.012, TRUE)

    CALL MontherAccount.issueBill(PowerInvoice)
    DISPLAY "Total Customer Outstanding Balance Owed: OMR " + CALL MontherAccount.calculateTotalOwedBalance()

END

    CLASS Route
        PRIVATE ATTRIBUTES:
            routeID : Integer
            waypointsList : List of String

        CONSTRUCTOR(id)
            SET this.routeID TO id
            SET this.waypointsList TO LIST []
        ENDCONSTRUCTOR

        METHOD addWaypoint(address)
            ADD address TO this.waypointsList
        ENDMETHOD

        METHOD getWaypointsCount()
            RETURN SIZE OF this.waypointsList
        ENDMETHOD
    ENDCLASS

    CLASS Ride
        PRIVATE ATTRIBUTES:
            rideID : Integer
            pickupLocation : String
            dropOffLocation : String
            distanceKM : Double
            rideRoute : Route

        CONSTRUCTOR(id, pickup, dropoff, distance, routeObj)
            SET this.rideID TO id
            SET this.pickupLocation TO pickup
            SET this.dropOffLocation TO dropoff
            SET this.distanceKM TO distance
            SET this.rideRoute TO routeObj
        ENDCONSTRUCTOR

        METHOD getDistanceKM()
            RETURN this.distanceKM
        ENDMETHOD

        ABSTRACT METHOD calculateFare()
    ENDCLASS

    CLASS StandardRide EXTENDS Ride
        PRIVATE ATTRIBUTES:
            ratePerKM : Double

        CONSTRUCTOR(id, pickup, dropoff, distance, routeObj, rate)
            CALL SUPER_CONSTRUCTOR(id, pickup, dropoff, distance, routeObj)
            SET this.ratePerKM TO rate
        ENDCONSTRUCTOR

        METHOD getRatePerKM()
            RETURN this.ratePerKM
        ENDMETHOD

        METHOD calculateFare()
            SET distance TO CALL this.getDistanceKM()
            RETURN distance * this.ratePerKM
        ENDMETHOD
    ENDCLASS

    CLASS SharedRide EXTENDS StandardRide
        PRIVATE ATTRIBUTES:
            numberOfPassengers : Integer

        CONSTRUCTOR(id, pickup, dropoff, distance, routeObj, rate, passengersCount)
            CALL SUPER_CONSTRUCTOR(id, pickup, dropoff, distance, routeObj, rate)
            SET this.numberOfPassengers TO passengersCount
        ENDCONSTRUCTOR

        METHOD calculateFare()
            SET standardTotal TO CALL SUPER_METHOD_calculateFare()
            IF this.numberOfPassengers Equals 0 THEN
                RETURN standardTotal
            ENDIF
            RETURN standardTotal / this.numberOfPassengers
        ENDMETHOD
    ENDCLASS

    CLASS PremiumRide EXTENDS Ride
        PRIVATE ATTRIBUTES:
            premiumRatePerKM : Double
            comfortSurcharge : Double

        CONSTRUCTOR(id, pickup, dropoff, distance, routeObj, pRate, charge)
            CALL SUPER_CONSTRUCTOR(id, pickup, dropoff, distance, routeObj)
            SET this.premiumRatePerKM TO pRate
            SET this.comfortSurcharge TO charge
        ENDCONSTRUCTOR

        METHOD calculateFare()
            SET distance TO CALL this.getDistanceKM()
            RETURN (distance * this.premiumRatePerKM) + this.comfortSurcharge
        ENDMETHOD
    ENDCLASS

    CREATE MyTripRoute FROM Route(77)
    CREATE Trip_1 FROM SharedRide(9942, "Al Badi", "Seeb Mall", 12.0, MyTripRoute, 0.200, 3)

    DISPLAY "Shared Ride Split Fare Cost Per Passenger: OMR " + CALL Trip_1.calculateFare()

END
