    CLASS Instrument
        PRIVATE ATTRIBUTES:
            brand : String
            price : Double

        CONSTRUCTOR(brnd, prc)
            SET this.brand TO brnd
            SET this.price TO prc
        ENDCONSTRUCTOR

        METHOD getBrand()
            RETURN this.brand
        ENDMETHOD

        METHOD getPrice()
            RETURN this.price
        ENDMETHOD

        METHOD describePlayStyle()
            RETURN "Generic Instrument Play Style"
        ENDMETHOD
    ENDCLASS

    CLASS Guitar EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfStrings : Integer

        CONSTRUCTOR(brnd, prc, strings)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfStrings TO strings
        ENDCONSTRUCTOR

        METHOD describePlayStyle()
            RETURN "strummed or plucked"
        ENDMETHOD
    ENDCLASS

    CLASS DrumKit EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfPieces : Integer

        CONSTRUCTOR(brnd, prc, pieces)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfPieces TO pieces
        ENDCONSTRUCTOR

        METHOD describePlayStyle()
            RETURN "struck"
        ENDMETHOD
    ENDCLASS

    CLASS Keyboard EXTENDS Instrument
        PRIVATE ATTRIBUTES:
            numberOfKeys : Integer

        CONSTRUCTOR(brnd, prc, keys)
            CALL SUPER_CONSTRUCTOR(brnd, prc)
            SET this.numberOfKeys TO keys
        ENDCONSTRUCTOR

        METHOD describePlayStyle()
            RETURN "pressed"
        ENDMETHOD
    ENDCLASS

    CLASS MusicStore
        PRIVATE ATTRIBUTES:
            storeID : Integer
            storeName : String
            instrumentsStock : List of Instrument

        CONSTRUCTOR(id, name)
            SET this.storeID TO id
            SET this.storeName TO name
            SET this.instrumentsStock TO LIST []
        ENDCONSTRUCTOR

        METHOD addInstrument(instrumentObj)
            ADD instrumentObj TO this.instrumentsStock
            DISPLAY "Instrument successfully added to the store stock inventory."
        ENDMETHOD

        METHOD calculateTotalStockValue()
            SET totalValue TO 0.0
            SET totalInstruments TO SIZE OF this.instrumentsStock
            
            REPEAT FROM i = 0 TO totalInstruments - 1
                SET totalValue TO totalValue + CALL this.instrumentsStock[i].getPrice()
            ENDREPEAT
            
            RETURN totalValue
        ENDMETHOD

        METHOD getStockCount()
            RETURN SIZE OF this.instrumentsStock
        ENDMETHOD
    ENDCLASS

    CREATE MyMusicStore FROM MusicStore(1, "Oman Harmony Sound")

    CREATE FenderGuitar FROM Guitar("Fender", 450.00, 6)
    CREATE YamahaKeyboard FROM Keyboard("Yamaha", 720.00, 88)

    CALL MyMusicStore.addInstrument(FenderGuitar)
    CALL MyMusicStore.addInstrument(YamahaKeyboard)

    DISPLAY "--- Store Live Dashboard Overview ---"
    DISPLAY "Total Active Instruments in Stock: " + CALL MyMusicStore.getStockCount()
    DISPLAY "Guitar Play Style Description: " + CALL FenderGuitar.describePlayStyle()
    DISPLAY "Keyboard Play Style Description: " + CALL YamahaKeyboard.describePlayStyle()
    DISPLAY "Total Inventory Financial Valuation: OMR " + CALL MyMusicStore.calculateTotalStockValue()

END

BEGIN

    CLASS Workout
        PRIVATE ATTRIBUTES:
            workoutDate : String
            durationMinutes : Integer

        CONSTRUCTOR(wDate, duration)
            SET this.workoutDate TO wDate
            SET this.durationMinutes TO duration
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS RunningWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            distanceKM : Double
            paceMinutesPerKM : Double

        CONSTRUCTOR(wDate, duration, dist, pc)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.distanceKM TO dist
            SET this.paceMinutesPerKM TO pc
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            RETURN distanceKM * 65.0
        ENDMETHOD
    ENDCLASS

    CLASS CyclingWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            distanceKM : Double
            resistanceLevel : Integer

        CONSTRUCTOR(wDate, duration, dist, resist)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.distanceKM TO dist
            SET this.resistanceLevel TO resist
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            RETURN distanceKM * (40.0 + resistanceLevel)
        ENDMETHOD
    ENDCLASS

    CLASS WeightTrainingWorkout EXTENDS Workout
        PRIVATE ATTRIBUTES:
            totalSets : Integer
            repsPerSet : Integer

        CONSTRUCTOR(wDate, duration, sets, reps)
            CALL SUPER_CONSTRUCTOR(wDate, duration)
            SET this.totalSets TO sets
            SET this.repsPerSet TO reps
        ENDCONSTRUCTOR

        METHOD calculateCalories()
            RETURN totalSets * repsPerSet * 0.4
        ENDMETHOD
    ENDCLASS

    CLASS FitnessLog
        PRIVATE ATTRIBUTES:
            logID : Integer
            workoutsList : List of Workout

        CONSTRUCTOR(id)
            SET this.logID TO id
            SET this.workoutsList TO LIST []
        ENDCONSTRUCTOR

        METHOD logWorkout(workoutObj)
            ADD workoutObj TO this.workoutsList
            DISPLAY "Workout session successfully saved to fitness log."
        ENDMETHOD

        METHOD calculateTotalCaloriesBurned()
            SET totalBurned TO 0.0
            SET totalSessions TO SIZE OF this.workoutsList
            
            REPEAT FROM i = 0 TO totalSessions - 1
                SET totalBurned TO totalBurned + CALL this.workoutsList[i].calculateCalories()
            ENDREPEAT
            
            RETURN totalBurned
        ENDMETHOD
    ENDCLASS

    CREATE MyFitnessLog FROM FitnessLog(1)
    
    CREATE Session_1 FROM RunningWorkout("2026-07-09", 45, 8.5, 5.2)
    CREATE Session_2 FROM WeightTrainingWorkout("2026-07-09", 60, 4, 12)

    CALL MyFitnessLog.logWorkout(Session_1)
    CALL MyFitnessLog.logWorkout(Session_2)

    DISPLAY "Live Fitness Dashboard Diagnostics"
    DISPLAY "Running Session Calories: " + CALL Session_1.calculateCalories()
    DISPLAY "Weight Training Session Calories: " + CALL Session_2.calculateCalories()
    DISPLAY "Total Cumulative Calories Burned: " + CALL MyFitnessLog.calculateTotalCaloriesBurned()

END
