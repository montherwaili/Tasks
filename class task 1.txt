        CONSTRUCTOR(type, limit)

        PRIVATE ATTRIBUTES:
            claimID : Integer
            status : String 
            approvedAmount : Double

        CONSTRUCTOR(id, amt)
            SET claimID = id
            SET approvedAmount = amt
            SET status = "Filed"
        ENDCONSTRUCTOR

    CLASS InsurancePolicy
        PRIVATE ATTRIBUTES:
            policyNumber : String
            holderName : String
            coverageDetails : Coverage
            claimsHistory : List of Claim

        CONSTRUCTOR(num, name, cvg)
            SET policyNumber = num
            SET holderName = name
            SET coverageDetails = cvg
            SET claimsHistory = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateRemainingCoverage()
            SET totalDeducted = 0.0
            SET totalClaims = SIZE OF claimsHistory
            
            REPEAT FROM i = 0 TO totalClaims - 1
                IF claimsHistory[i].status == "Approved" THEN
                    SET totalDeducted = totalDeducted + claimsHistory[i].approvedAmount
                ENDIF
            ENDREPEAT
            
            RETURN coverageDetails.maxLimit - totalDeducted
        ENDMETHOD

        METHOD fileClaim(claimObj)
            IF claimObj.approvedAmount > coverageDetails.maxLimit THEN
                DISPLAY "Claim Refused: Amount exceeds individual coverage limits!"
                SET claimObj.status = "Rejected"
                RETURN FALSE
            ENDIF
            
            SET claimObj.status = "Approved"
            ADD claimObj TO claimsHistory
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

                ENDIF

            SET activitiesLog = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD logActivity(activityObj)
            IF CALL activityObj.assignedWorker.canAcceptWork() == TRUE THEN
                ADD activityObj TO activitiesLog
                CALL activityObj.assignedWorker.incrementDailyWork()
                DISPLAY "Activity successfully recorded on the field log."
                RETURN TRUE
            ELSE
                DISPLAY "Scheduling Refused: Worker has hit their daily assignment limit!"
                RETURN FALSE
            ENDIF
        ENDMETHOD

        METHOD estimateYield()
            SET logSize = SIZE OF activitiesLog
            SET careScore = 0.0
            
            REPEAT FROM i = 0 TO logSize - 1
                IF CALL activitiesLog[i].getActivityType() == "Watering" THEN
                    SET careScore = careScore + 0.5
                ENDIF
                IF CALL activitiesLog[i].getActivityType() == "Fertilizing" THEN
                    SET careScore = careScore + 0.5
                ENDIF
            ENDREPEAT

            
            IF careScore == 0.0 THEN
                RETURN CALL plantedCrop.getExpectedOutput() * 0.4 
            ELSE
                IF careScore > 1.5 THEN
                    SET careScore = 1.5
                ENDIF
                RETURN CALL plantedCrop.getExpectedOutput() * careScore
            ENDIF
        ENDMETHOD
    ENDCLASS

    CREATE WheatCrop FROM Crop("Golden Wheat", 500.00, 3)
    CREATE FarmWorker FROM Worker(12, "Salem Al-Harthy", 1) 
    CREATE Field_North FROM Field(701, WheatCrop)

    CREATE Act_1 FROM FarmActivity("Watering", FarmWorker)
    CREATE Act_2 FROM FarmActivity("Fertilizing", FarmWorker) 

    CALL Field_North.logActivity(Act_1)
    CALL Field_North.logActivity(Act_2) 

    DISPLAY "Calculated Scientific Yield Estimate: " + CALL Field_North.estimateYield() + " KG"

END
