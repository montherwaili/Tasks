START
BEGIN

 
    CLASS Species
        PRIVATE ATTRIBUTES:
            speciesName : String
            careRequirements : String

        CONSTRUCTOR(name, reqs)
            SET speciesName = name
            SET careRequirements = reqs
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Treatment
        PRIVATE ATTRIBUTES:
            treatmentID : Integer
            treatmentName : String
            cost : Double

        CONSTRUCTOR(id, name, price)
            SET treatmentID = id
            SET treatmentName = name
            SET cost = price
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Examination
        PRIVATE ATTRIBUTES:
            examID : Integer
            diagnosis : String
            heartRate : Integer       
            temperature : Double      
            isUrgent : Boolean
            treatmentsPrescribed : List of Treatment

        CONSTRUCTOR(id, diag, hr, temp)
            SET examID = id
            SET diagnosis = diag
            SET heartRate = hr
            SET temperature = temp
            SET isUrgent = FALSE 
            SET treatmentsPrescribed = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD checkVitals()
            IF heartRate < 60 OR heartRate > 140 OR temperature < 38.0 OR temperature > 39.2 THEN
                SET isUrgent = TRUE
            ELSE
                SET isUrgent = FALSE
            ENDIF
        ENDMETHOD

        METHOD addTreatment(treatmentObject)
            ADD treatmentObject TO treatmentsPrescribed
        ENDMETHOD

        METHOD getVisitTotal()
            SET total = 0.0
            SET listSize = SIZE OF treatmentsPrescribed
            
            REPEAT FROM i = 0 TO listSize - 1
                SET currentTreatment = treatmentsPrescribed[i]
                SET total = total + currentTreatment.cost
            ENDREPEAT
            
            RETURN total
        ENDMETHOD
    ENDCLASS

    CLASS Pet
        PRIVATE ATTRIBUTES:
            petID : Integer
            name : String
            dateOfBirth : Integer     
            speciesInfo : Species
            medicalHistory : List of Examination

        CONSTRUCTOR(id, petName, birthYear, speciesObj)
            SET petID = id
            SET name = petName
            SET dateOfBirth = birthYear
            SET speciesInfo = speciesObj
            SET medicalHistory = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateAge(currentYear)
            RETURN currentYear - dateOfBirth
        ENDMETHOD

        METHOD addExaminationToHistory(examObject)
            ADD examObject TO medicalHistory
        ENDMETHOD
    ENDCLASS

    CLASS Owner
        PRIVATE ATTRIBUTES:
            ownerID : Integer
            name : String
            phoneNumber : String
            petsList : List of Pet

        CONSTRUCTOR(id, ownerName, phone)
            SET ownerID = id
            SET name = ownerName
            SET phoneNumber = phone
            SET petsList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addPet(petObject)
            ADD petObject TO petsList
        ENDMETHOD
    ENDCLASS


    CREATE CatSpecies FROM Species("Feline (Cat)", "Requires annual vaccination")

    CREATE Treat_1 FROM Treatment(901, "Antibiotic Injection", 15.50)
    CREATE Treat_2 FROM Treatment(902, "IV Fluid Therapy", 25.00)

    CREATE Exam_1 FROM Examination(4401, "Severe Infection", 155, 40.1)
    
    CALL Exam_1.addTreatment(Treat_1)
    CALL Exam_1.addTreatment(Treat_2)
    
    CALL Exam_1.checkVitals()

    CREATE Pet_Leo FROM Pet(5510, "Leo", 2022, CatSpecies)
    CALL Pet_Leo.addExaminationToHistory(Exam_1)

    CREATE Owner_Monther FROM Owner(1002, "Monther Al-Waili", "+96899998888")
    CALL Owner_Monther.addPet(Pet_Leo)


    DISPLAY "--- Veterinary Clinic Visit Summary (Repeat From Mode) ---"
    DISPLAY "Pet Owner: " + Owner_Monther.name
    DISPLAY "Pet Name: " + Pet_Leo.name + " | Species: " + Pet_Leo.speciesInfo.speciesName
    DISPLAY "Derived Pet Age: " + CALL Pet_Leo.calculateAge(2026) + " years old"
    DISPLAY "--- Examination Details ---"
    DISPLAY "Diagnosis: " + Exam_1.diagnosis
    DISPLAY "Is Visit Urgent? : " + Exam_1.isUrgent
    DISPLAY "Total Dynamic Prescribed Bill: OMR " + CALL Exam_1.getVisitTotal()

END

BEGIN
    CLASS Screen
        PRIVATE ATTRIBUTES:
            screenID : Integer
            seatingLayout : String
            fixedCapacity : Integer

        CONSTRUCTOR(id, layout, capacity)
            SET screenID = id
            SET seatingLayout = layout
            SET fixedCapacity = capacity
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Film
        PRIVATE ATTRIBUTES:
            filmID : Integer
            title : String
            durationInMinutes : Integer
            genre : String

        CONSTRUCTOR(id, tlt, dur, gnr)
            SET filmID = id
            SET title = tlt
            SET durationInMinutes = dur
            SET genre = gnr
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Ticket
        PRIVATE ATTRIBUTES:
            ticketID : Integer
            seatNumber : String
            price : Double

        CONSTRUCTOR(id, seat, prc)
            SET ticketID = id
            SET seatNumber = seat
            SET price = prc
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Showtime
        PRIVATE ATTRIBUTES:
            showtimeID : Integer
            filmInfo : Film
            screenInfo : Screen
            startTime : Integer 
            endTime : Integer
            bookedTickets : List of Ticket

        CONSTRUCTOR(id, flm, scr, start, end)
            SET showtimeID = id
            SET filmInfo = flm
            SET screenInfo = scr
            SET startTime = start
            SET endTime = end
            SET bookedTickets = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateAvailableSeats()
            SET totalBooked = SIZE OF bookedTickets
            RETURN screenInfo.fixedCapacity - totalBooked
        ENDMETHOD

        METHOD bookTicket(ticketObject)
            SET totalBooked = SIZE OF bookedTickets
            IF totalBooked >= screenInfo.fixedCapacity THEN
                DISPLAY "Error: Screening room is full!"
                RETURN FALSE
            ENDIF

            SET isSeatTaken = FALSE
            REPEAT FROM i = 0 TO totalBooked - 1
                IF bookedTickets[i].seatNumber == ticketObject.seatNumber THEN
                    SET isSeatTaken = TRUE
                ENDIF
            ENDREPEAT

            IF isSeatTaken == TRUE THEN
                DISPLAY "Error: Seat already booked!"
                RETURN FALSE
            ENDIF

            ADD ticketObject TO bookedTickets
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CREATE CinemaScreen FROM Screen(1, "Standard IMAX", 5)
    CREATE Movie_Dune FROM Film(501, "Dune: Part Two", 166, "Sci-Fi")
    CREATE Show_1 FROM Showtime(9001, Movie_Dune, CinemaScreen, 1800, 2100)

    CREATE Tkt_1 FROM Ticket(11, "A1", 4.50)
    CREATE Tkt_2 FROM Ticket(12, "A1", 4.50) 

    CALL Show_1.bookTicket(Tkt_1)
    CALL Show_1.bookTicket(Tkt_2) 

    DISPLAY "Remaining Available Seats: " + CALL Show_1.calculateAvailableSeats()
END

BEGIN
    CLASS Vendor
        PRIVATE ATTRIBUTES:
            vendorID : Integer
            name : String
            reliabilityRating : Double

        CONSTRUCTOR(id, vName, rating)
            SET vendorID = id
            SET name = vName
            SET reliabilityRating = rating
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS StorageBin
        PRIVATE ATTRIBUTES:
            binID : Integer
            locationCode : String
            currentQty : Integer

        CONSTRUCTOR(id, code, qty)
            SET binID = id
            SET locationCode = code
            SET currentQty = qty
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Product
        PRIVATE ATTRIBUTES:
            productID : Integer
            name : String
            vendorInfo : Vendor
            binsList : List of StorageBin

        CONSTRUCTOR(id, pName, vnd)
            SET productID = id
            SET name = pName
            SET vendorInfo = vnd
            SET binsList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addStorageBin(binObj)
            ADD binObj TO binsList
        ENDMETHOD

        METHOD getTotalStock()
            SET sum = 0
            SET totalBins = SIZE OF binsList
            REPEAT FROM i = 0 TO totalBins - 1
                SET sum = sum + binsList[i].currentQty
            ENDREPEAT
            RETURN sum
        ENDMETHOD

        METHOD processOutgoingStock(qtyToMove)
            SET currentStock = CALL getTotalStock()
            IF currentStock - qtyToMove < 0 THEN
                DISPLAY "Transaction Rejected: Insufficient warehouse stock!"
                RETURN FALSE
            ELSE
                SET binsList[0].currentQty = binsList[0].currentQty - qtyToMove
                DISPLAY "Stock movement approved successfully."
                RETURN TRUE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CREATE Supplier FROM Vendor(77, "Tech Logistics", 94.5)
    CREATE Pro_Keyboard FROM Product(101, "Mechanical Keyboard", Supplier)
    
    CREATE Bin_A FROM StorageBin(1, "A-10", 50)
    CREATE Bin_B FROM StorageBin(2, "B-12", 30)
    
    CALL Pro_Keyboard.addStorageBin(Bin_A)
    CALL Pro_Keyboard.addStorageBin(Bin_B)

    DISPLAY "Initial Total Warehouse Stock: " + CALL Pro_Keyboard.getTotalStock()
    CALL Pro_Keyboard.processOutgoingStock(90) 
END
STOP

START
BEGIN
    CLASS Grade
        PRIVATE ATTRIBUTES:
            assessmentName : String
            score : Double
            weight : Double 

        CONSTRUCTOR(name, scr, wgt)
            SET assessmentName = name
            SET score = scr
            SET weight = wgt
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Student
        PRIVATE ATTRIBUTES:
            studentID : Integer
            name : String
            gradesList : List of Grade

        CONSTRUCTOR(id, sName)
            SET studentID = id
            SET name = sName
            SET gradesList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addGrade(gradeObj)
            ADD gradeObj TO gradesList
        ENDMETHOD

        METHOD calculateOverallAverage()
            SET totalPoints = 0.0
            SET totalWeights = 0.0
            SET totalGrades = SIZE OF gradesList
            
            IF totalGrades == 0 THEN
                RETURN 0.0
            ENDIF

            REPEAT FROM i = 0 TO totalGrades - 1
                SET totalPoints = totalPoints + (gradesList[i].score * gradesList[i].weight)
                SET totalWeights = totalWeights + gradesList[i].weight
            ENDREPEAT
            
            RETURN totalPoints / totalWeights
        ENDMETHOD
    ENDCLASS

    CLASS SchoolClass
        PRIVATE ATTRIBUTES:
            classID : String
            subjectName : String
            maxSize : Integer
            roster : List of Student

        CONSTRUCTOR(id, sub, maxNum)
            SET classID = id
            SET subjectName = sub
            SET maxSize = maxNum
            SET roster = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD enrollStudent(studentObj)
            SET currentEnrolled = SIZE OF roster
            IF currentEnrolled >= maxSize THEN
                DISPLAY "Enrollment Terminated: Class capacity exceeded!"
                RETURN FALSE
            ENDIF
            ADD studentObj TO roster
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CREATE MathClass FROM SchoolClass("MATH-101", "Calculus", 2)
    CREATE Std_Monther FROM Student(7015, "Monther Al-Waili")
    
    CREATE Mark_1 FROM Grade("Midterm Exam", 90.0, 0.40)
    CREATE Mark_2 FROM Grade("Final Project", 95.0, 0.60)
    
    CALL Std_Monther.addGrade(Mark_1)
    CALL Std_Monther.addGrade(Mark_2)

    DISPLAY "Calculated Weighted Academic Average: " + CALL Std_Monther.calculateOverallAverage()
END
STOP

START
BEGIN
    CLASS Coverage
        PRIVATE ATTRIBUTES:
            coverageType : String
            maxLimit : Double

        CONSTRUCTOR(type, limit)
            SET coverageType = type
            SET maxLimit = limit
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Claim
        PRIVATE ATTRIBUTES:
            claimID : Integer
            status : String 
            approvedAmount : Double

        CONSTRUCTOR(id, amt)
            SET claimID = id
            SET approvedAmount = amt
            SET status = "Filed"
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS InsurancePolicy
        PRIVATE ATTRIBUTES:
            policyNumber : String
            holderName : String
            coverageDetails : Coverage
            claimsHistory : List of Claim

        CONSTRUCTOR(num, name, cvg)
            SET policyNumber = num
            SET holderName = name
            SET coverageDetails = cvg
            SET claimsHistory = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateRemainingCoverage()
            SET totalDeducted = 0.0
            SET totalClaims = SIZE OF claimsHistory
            
            REPEAT FROM i = 0 TO totalClaims - 1
                IF claimsHistory[i].status == "Approved" THEN
                    SET totalDeducted = totalDeducted + claimsHistory[i].approvedAmount
                ENDIF
            ENDREPEAT
            
            RETURN coverageDetails.maxLimit - totalDeducted
        ENDMETHOD

        METHOD fileClaim(claimObj)
            IF claimObj.approvedAmount > coverageDetails.maxLimit THEN
                DISPLAY "Claim Refused: Amount exceeds individual coverage limits!"
                SET claimObj.status = "Rejected"
                RETURN FALSE
            ENDIF
            
            SET claimObj.status = "Approved"
            ADD claimObj TO claimsHistory
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CREATE CarCoverage FROM Coverage("Auto Collision", 5000.00)
    CREATE MyPolicy FROM InsurancePolicy("POL-9942", "Monther Al-Waili", CarCoverage)
    
    CREATE AccidentClaim FROM Claim(7701, 1200.00)
    CALL MyPolicy.fileClaim(AccidentClaim)

    DISPLAY "Remaining Trustworthy Policy Budget: OMR " + CALL MyPolicy.calculateRemainingCoverage()
END
STOP
