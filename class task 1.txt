        CONSTRUCTOR(id, flm, scr, start, end)

            SET totalBooked = SIZE OF bookedTickets

        METHOD bookTicket(ticketObject)
            SET totalBooked = SIZE OF bookedTickets
            SET maxCapacity = CALL screenInfo.getFixedCapacity()
            
            IF totalBooked >= maxCapacity THEN
                DISPLAY "Error: Screening room is full!"
                RETURN FALSE
            ENDIF

            SET isSeatTaken = FALSE
            REPEAT FROM i = 0 TO totalBooked - 1
                SET currentBookedSeat = CALL bookedTickets[i].getSeatNumber()
                SET targetBookingSeat = CALL ticketObject.getSeatNumber()
                
                IF currentBookedSeat == targetBookingSeat THEN
                    SET isSeatTaken = TRUE
                ENDIF
            ENDREPEAT

            IF isSeatTaken == TRUE THEN
                DISPLAY "Error: Seat already booked!"
                RETURN FALSE
            ENDIF

            ADD ticketObject TO bookedTickets
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

        PRIVATE ATTRIBUTES:

        CONSTRUCTOR(id)
            SET scheduleID = id
            SET allShowtimes = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addShowtime(newShowtimeObj)
            SET totalShows = SIZE OF allShowtimes
            SET targetScreenID = CALL CALL newShowtimeObj.getScreenInfo().getScreenID()
            SET newStart = CALL newShowtimeObj.getStartTime()
            SET newEnd = CALL newShowtimeObj.getEndTime()

            REPEAT FROM i = 0 TO totalShows - 1
                SET existingShow = allShowtimes[i]
                SET existingScreenID = CALL CALL existingShow.getScreenInfo().getScreenID()
                
                IF existingScreenID == targetScreenID THEN
                    SET extStart = CALL existingShow.getStartTime()
                    SET extEnd = CALL existingShow.getEndTime()
                    
                    IF newStart < extEnd AND newEnd > extStart THEN
                        DISPLAY "Scheduling Rejected: Showtime overlaps with an existing movie on Screen " + targetScreenID
                        RETURN FALSE
                    ENDIF
                ENDIF
            ENDREPEAT

            ADD newShowtimeObj TO allShowtimes
            DISPLAY "Showtime successfully scheduled and verified."
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CLASS Product
        PRIVATE ATTRIBUTES:
            productID : Integer
            name : String
            vendorInfo : Vendor
            binsList : List of StorageBin

        CONSTRUCTOR(id, pName, vnd)
            SET productID = id
            SET name = pName
            SET vendorInfo = vnd
            SET binsList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addStorageBin(binObj)
            ADD binObj TO binsList
        ENDMETHOD

        METHOD getTotalStock()
            SET sum = 0
            SET totalBins = SIZE OF binsList
            REPEAT FROM i = 0 TO totalBins - 1
                SET sum = sum + CALL binsList[i].getCurrentQty()
            ENDREPEAT
            RETURN sum
        ENDMETHOD

        METHOD processOutgoingStock(qtyToMove)
            SET currentStock = CALL getTotalStock()
            
            IF currentStock - qtyToMove < 0 THEN
                DISPLAY "Transaction Rejected: Insufficient warehouse stock!"
                RETURN FALSE
            ENDIF

            SET remainingToMove = qtyToMove
            SET totalBins = SIZE OF binsList
            
            REPEAT FROM i = 0 TO totalBins - 1
                IF remainingToMove > 0 THEN
                    SET binQty = CALL binsList[i].getCurrentQty()
                    
                    IF binQty >= remainingToMove THEN
                        CALL binsList[i].setCurrentQty(binQty - remainingToMove)
                        SET remainingToMove = 0
                    ELSE
                        SET remainingToMove = remainingToMove - binQty
                        CALL binsList[i].setCurrentQty(0)
                    ENDIF
                ENDIF
            ENDREPEAT

            DISPLAY "Stock movement approved and distributed across bins successfully."
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

        CONSTRUCTOR(id, sub, time, maxNum)
            SET classID = id
            SET subjectName = sub

            SET maxSize = maxNum
            SET roster = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD enrollStudent(studentObj)
            SET currentEnrolled = CALL getRosterSize()
            SET maxCap = CALL getMaxSize()
            
            IF currentEnrolled >= maxCap THEN
                DISPLAY "Enrollment Terminated: Class capacity exceeded!"
                RETURN FALSE
            ENDIF

            SET targetTime = scheduleTime
            IF CALL studentObj.hasScheduleConflict(targetTime) == TRUE THEN
                DISPLAY "Enrollment Terminated: Student has a schedule conflict at " + targetTime
                RETURN FALSE
            ENDIF

            ADD studentObj TO roster
            CALL studentObj.addClass(THIS)
            DISPLAY "Student successfully enrolled in " + classID
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CLASS Student
        PRIVATE ATTRIBUTES:
            studentID : Integer
            name : String
            gradesList : List of Grade
            enrolledClasses : List of SchoolClass 

        CONSTRUCTOR(id, sName)
            SET studentID = id
            SET name = sName
            SET gradesList = INITIALIZE EMPTY LIST
            SET enrolledClasses = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addGrade(gradeObj)
            ADD gradeObj TO gradesList
        ENDMETHOD

        METHOD addClass(classObj)
            ADD classObj TO enrolledClasses
        ENDMETHOD

        METHOD hasScheduleConflict(targetTime)
            SET totalClasses = SIZE OF enrolledClasses
            REPEAT FROM i = 0 TO totalClasses - 1
                SET currentClassTime = CALL enrolledClasses[i].getScheduleTime()
                IF currentClassTime == targetTime THEN
                    RETURN TRUE 
                ENDIF
            ENDREPEAT
            RETURN FALSE
        ENDMETHOD

        METHOD calculateOverallAverage()
            SET totalPoints = 0.0
            SET totalWeights = 0.0
            SET totalGrades = SIZE OF gradesList
            
            IF totalGrades == 0 THEN
                RETURN 0.0
            ENDIF

            REPEAT FROM i = 0 TO totalGrades - 1
                SET currentScore = CALL gradesList[i].getScore()
                SET currentWeight = CALL gradesList[i].getWeight()
                SET totalPoints = totalPoints + (currentScore * currentWeight)
                SET totalWeights = totalWeights + currentWeight
            ENDREPEAT
            
            RETURN totalPoints / totalWeights
        ENDMETHOD
    ENDCLASS

            SET statusStage = 1

            RETURN approvedAmount

    CLASS InsurancePolicy
        PRIVATE ATTRIBUTES:
            policyNumber : String
            holderName : String
            coverageDetails : Coverage
            claimsHistory : List of Claim

        CONSTRUCTOR(num, name, cvg)
            SET policyNumber = num
            SET holderName = name
            SET coverageDetails = cvg
            SET claimsHistory = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateRemainingCoverage()
            SET totalDeducted = 0.0
            SET totalClaims = SIZE OF claimsHistory
            
            REPEAT FROM i = 0 TO totalClaims - 1
                IF CALL claimsHistory[i].getStatusStage() >= 3 THEN
                    SET totalDeducted = totalDeducted + CALL claimsHistory[i].getApprovedAmount()
                ENDIF
            ENDREPEAT
            
            RETURN CALL coverageDetails.getMaxLimit() - totalDeducted
        ENDMETHOD

        METHOD fileClaim(claimObj)
            IF CALL claimObj.getApprovedAmount() > CALL coverageDetails.getMaxLimit() THEN
                DISPLAY "Claim Refused: Amount exceeds individual coverage limits!"
                RETURN FALSE
            ENDIF
            
            ADD claimObj TO claimsHistory
            DISPLAY "Claim successfully filed and logged under policy."
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

        PRIVATE ATTRIBUTES:
            eventID : Integer
            coupleName : String
            budget : Double
            eventDate : String
            tasksList : List of PlanningTask
            guestList : List of Guest

        CONSTRUCTOR(id, couple, bgt, eDate)

            SET eventDate = eDate
            SET tasksList = INITIALIZE EMPTY LIST

        METHOD addTask(taskObj)
            SET currentSpending = 0.0
            SET totalTasks = SIZE OF tasksList
            
            REPEAT FROM i = 0 TO totalTasks - 1
                SET currentSpending = currentSpending + CALL tasksList[i].getCost()
            ENDREPEAT

            IF currentSpending + CALL taskObj.getCost() > budget THEN
                DISPLAY "Error: Task rejected! Spending exceeds the set budget limit."
                RETURN FALSE
            ENDIF

            SET targetVendor = CALL taskObj.getAssignedVendor()
            IF targetVendor != NULL THEN
                SET requestedDate = CALL taskObj.getTaskDate()
                IF CALL targetVendor.isAvailable(requestedDate) == FALSE THEN
                    DISPLAY "Error: Vendor double-booking prevented for this date!"
                    RETURN FALSE
                ENDIF
                CALL targetVendor.bookDate(requestedDate)
            ENDIF

            ADD taskObj TO tasksList
            RETURN TRUE
        ENDMETHOD

            SET totalTasks = SIZE OF tasksList
            IF totalTasks == 0 THEN
                RETURN 0.0
            ENDIF

            SET doneCount = 0
            REPEAT FROM i = 0 TO totalTasks - 1
                IF CALL tasksList[i].getIsDone() == TRUE THEN
                    SET doneCount = doneCount + 1
                ENDIF
            ENDREPEAT

            RETURN (doneCount / totalTasks) * 100.0
        ENDMETHOD

            SET totalSold = SIZE OF soldTicketsList
            SET isSeatTaken = FALSE
            
            REPEAT FROM i = 0 TO totalSold - 1
                IF CALL soldTicketsList[i].getSeatCode() == CALL newTicketObj.getSeatCode() THEN
                    SET isSeatTaken = TRUE
                ENDIF
            ENDREPEAT

            IF isSeatTaken == TRUE THEN
                DISPLAY "Error: Transaction Failed! Seat is already reserved."
                RETURN FALSE
            ENDIF

            ADD newTicketObj TO soldTicketsList
            RETURN TRUE
        ENDMETHOD

            ADD concertObj TO concertsList
            DISPLAY "Concert successfully scheduled."
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

            stage : Integer

        CONSTRUCTOR(id, tlt, devObj)

            RETURN assignedDev

        METHOD advanceStage()
            IF stage == 4 THEN
                DISPLAY "Task Error: Task is already in the final Complete stage!"
                RETURN FALSE

            SET stage = stage + 1
            DISPLAY "Task stage successfully advanced to stage: " + stage
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CLASS Sprint
        PRIVATE ATTRIBUTES:
            sprintID : Integer
            sprintName : String
            sprintTasksList : List of Task

        CONSTRUCTOR(id, name)
            SET sprintID = id
            SET sprintName = name
            SET sprintTasksList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addTaskToSprint(taskObj)
            SET targetDev = CALL taskObj.getAssignedDev()
            IF CALL targetDev.hasCapacity() == TRUE THEN
                ADD taskObj TO sprintTasksList
                CALL targetDev.incrementWorkload()
                RETURN TRUE
            ELSE
                DISPLAY "Sprint Error: Developer capacity exceeded for this sprint!"
                RETURN FALSE
            ENDIF
        ENDMETHOD

        METHOD calculateProgress()
            SET totalTasks = SIZE OF sprintTasksList
            IF totalTasks == 0 THEN
                RETURN 0.0
            ENDIF

            SET completedCount = 0
            REPEAT FROM i = 0 TO totalTasks - 1
                IF CALL sprintTasksList[i].getStage() == 4 THEN
                    SET completedCount = completedCount + 1
                ENDIF
            ENDREPEAT

            RETURN (completedCount / totalTasks) * 100.0
        ENDMETHOD
    ENDCLASS

    CLASS Field
        PRIVATE ATTRIBUTES:
            fieldID : Integer
            plantedCrop : Crop
            activitiesLog : List of FarmActivity

        CONSTRUCTOR(id, crp)
            SET fieldID = id
            SET plantedCrop = crp
            SET activitiesLog = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD logActivity(activityObj)
            SET targetWorker = CALL activityObj.getAssignedWorker()
            
            IF CALL targetWorker.canAcceptWork() == FALSE THEN
                DISPLAY "Scheduling Refused: Worker has hit their daily assignment limit!"
                RETURN FALSE
            ENDIF

            IF CALL activityObj.getActivityType() == "Watering" THEN
                SET providedWater = CALL activityObj.getWaterAmount()
                SET requiredWater = CALL plantedCrop.getWaterRequirement()
                
                IF providedWater < requiredWater THEN
                    DISPLAY "Activity Refused: Provided water does not meet crop requirement!"
                    RETURN FALSE
                ENDIF
            ENDIF

            ADD activityObj TO activitiesLog
            CALL targetWorker.incrementDailyWork()
            DISPLAY "Activity successfully recorded on the field log."
            RETURN TRUE
        ENDMETHOD

        METHOD estimateYield()
            SET logSize = SIZE OF activitiesLog
            SET careScore = 0.0
            
            REPEAT FROM i = 0 TO logSize - 1
                IF CALL activitiesLog[i].getActivityType() == "Watering" THEN
                    SET careScore = careScore + 0.5
                ENDIF
                IF CALL activitiesLog[i].getActivityType() == "Fertilizing" THEN
                    SET careScore = careScore + 0.5
                ENDIF
            ENDREPEAT

            IF careScore == 0.0 THEN
                RETURN CALL plantedCrop.getExpectedOutput() * 0.4 
            ELSE
                IF careScore > 1.5 THEN
                    SET careScore = 1.5
                ENDIF
                RETURN CALL plantedCrop.getExpectedOutput() * careScore
            ENDIF
        ENDMETHOD
    ENDCLASS
