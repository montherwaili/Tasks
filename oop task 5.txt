    CLASS Shape
        PRIVATE ATTRIBUTES:
            color : String
            positionX : Integer
            positionY : Integer

        CONSTRUCTOR(clr, x, y)
            SET color = clr
            SET positionX = x
            SET positionY = y
        ENDCONSTRUCTOR

        METHOD getColor()
            RETURN color
        ENDMETHOD

        METHOD calculateArea()
            RETURN 0.0
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS Circle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            radius : Double

        CONSTRUCTOR(clr, x, y, r)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET radius = r
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN 3.14 * radius * radius
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * 3.14 * radius
        ENDMETHOD
    ENDCLASS

    CLASS Rectangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            width : Double
            height : Double

        CONSTRUCTOR(clr, x, y, w, h)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET width = w
            SET height = h
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN width * height
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * (width + height)
        ENDMETHOD
    ENDCLASS

    CLASS Triangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            sideA : Double
            sideB : Double
            sideC : Double

        CONSTRUCTOR(clr, x, y, a, b, c)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET sideA = a
            SET sideB = b
            SET sideC = c
        ENDCONSTRUCTOR

        METHOD calculateArea()
            SET s = (sideA + sideB + sideC) / 2.0
            RETURN CALL MATH_SQRT(s * (s - sideA) * (s - sideB) * (s - sideC))
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN sideA + sideB + sideC
        ENDMETHOD
    ENDCLASS

    CLASS Canvas
        PRIVATE ATTRIBUTES:
            canvasID : Integer
            shapesCollection : List of Shape

        CONSTRUCTOR(id)
            SET canvasID = id
            SET shapesCollection = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addShape(shapeObj)
            ADD shapeObj TO shapesCollection
            DISPLAY "Shape successfully drawn on the canvas."
        ENDMETHOD

        METHOD getShapesCount()
            RETURN SIZE OF shapesCollection
        ENDMETHOD
    ENDCLASS

    CREATE MyCanvas FROM Canvas(1)

    CREATE RedCircle FROM Circle("Red", 10, 20, 5.0)
    CREATE BlueRectangle FROM Rectangle("Blue", 0, 0, 4.0, 6.0)

    CALL MyCanvas.addShape(RedCircle)
    CALL MyCanvas.addShape(BlueRectangle)

    DISPLAY "Total Shapes on Canvas: " + CALL MyCanvas.getShapesCount()
    DISPLAY "Circle Color: " + CALL RedCircle.getColor()
    DISPLAY "Circle Area: " + CALL RedCircle.calculateArea()
    DISPLAY "Rectangle Perimeter: " + CALL BlueRectangle.calculatePerimeter()

END

    CLASS Employee
        PRIVATE ATTRIBUTES:
            name : String
            employeeID : Integer
            hireDate : String

        CONSTRUCTOR(empName, id, hDate)
            SET name = empName
            SET employeeID = id
            SET hireDate = hDate
        ENDCONSTRUCTOR

        METHOD getName()
            RETURN name
        ENDMETHOD

        METHOD calculatePay()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS SalariedEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            fixedMonthlySalary : Double

        CONSTRUCTOR(empName, id, hDate, salary)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET fixedMonthlySalary = salary
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN fixedMonthlySalary
        ENDMETHOD
    ENDCLASS

    CLASS HourlyEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            hoursWorked : Integer
            hourlyRate : Double

        CONSTRUCTOR(empName, id, hDate, hours, rate)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET hoursWorked = hours
            SET hourlyRate = rate
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN hoursWorked * hourlyRate
        ENDMETHOD
    ENDCLASS

    CLASS CommissionEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            basePay : Double
            salesAmount : Double
            commissionRate : Double

        CONSTRUCTOR(empName, id, hDate, bPay, sales, cRate)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET basePay = bPay
            SET salesAmount = sales
            SET commissionRate = cRate
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN basePay + (salesAmount * commissionRate)
        ENDMETHOD
    ENDCLASS

    CLASS Payroll
        PRIVATE ATTRIBUTES:
            payrollID : Integer
            employeesList : List of Employee

        CONSTRUCTOR(id)
            SET payrollID = id
            SET employeesList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addEmployee(empObj)
            ADD empObj TO employeesList
            DISPLAY "Employee successfully added to payroll system."
        ENDMETHOD

        METHOD calculateTotalPayout()
            SET totalPayout = 0.0
            SET totalEmployees = SIZE OF employeesList
            
            REPEAT FROM i = 0 TO totalEmployees - 1
                SET totalPayout = totalPayout + CALL employeesList[i].calculatePay()
            ENDREPEAT
            
            RETURN totalPayout
        ENDMETHOD
    ENDCLASS

    CREATE CompanyPayroll FROM Payroll(1)

    CREATE Emp_1 FROM SalariedEmployee("Monther Al-Waili", 7015, "2026-01-01", 1200.00)
    CREATE Emp_2 FROM HourlyEmployee("Ahmed Al-Balushi", 8022, "2026-03-15", 160, 5.00)

    CALL CompanyPayroll.addEmployee(Emp_1)
    CALL CompanyPayroll.addEmployee(Emp_2)

    DISPLAY "Payroll System Live Overview"
    DISPLAY "Employee 1 Name: " + CALL Emp_1.getName()
    DISPLAY "Employee 1 Calculated Pay: OMR " + CALL Emp_1.calculatePay()
    DISPLAY "Employee 2 Calculated Pay: OMR " + CALL Emp_2.calculatePay()
    DISPLAY "Total Corporate Payroll Payout: OMR " + CALL CompanyPayroll.calculateTotalPayout()

END
