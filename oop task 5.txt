    CLASS Shape
        PRIVATE ATTRIBUTES:
            color : String
            positionX : Integer
            positionY : Integer

        CONSTRUCTOR(clr, x, y)
            SET color = clr
            SET positionX = x
            SET positionY = y
        ENDCONSTRUCTOR

        METHOD getColor()
            RETURN color
        ENDMETHOD

        METHOD calculateArea()
            RETURN 0.0
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS Circle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            radius : Double

        CONSTRUCTOR(clr, x, y, r)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET radius = r
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN 3.14 * radius * radius
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * 3.14 * radius
        ENDMETHOD
    ENDCLASS

    CLASS Rectangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            width : Double
            height : Double

        CONSTRUCTOR(clr, x, y, w, h)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET width = w
            SET height = h
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN width * height
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * (width + height)
        ENDMETHOD
    ENDCLASS

    CLASS Triangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            sideA : Double
            sideB : Double
            sideC : Double

        CONSTRUCTOR(clr, x, y, a, b, c)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET sideA = a
            SET sideB = b
            SET sideC = c
        ENDCONSTRUCTOR

        METHOD calculateArea()
            SET s = (sideA + sideB + sideC) / 2.0
            RETURN CALL MATH_SQRT(s * (s - sideA) * (s - sideB) * (s - sideC))
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN sideA + sideB + sideC
        ENDMETHOD
    ENDCLASS

    CLASS Canvas
        PRIVATE ATTRIBUTES:
            canvasID : Integer
            shapesCollection : List of Shape

        CONSTRUCTOR(id)
            SET canvasID = id
            SET shapesCollection = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addShape(shapeObj)
            ADD shapeObj TO shapesCollection
            DISPLAY "Shape successfully drawn on the canvas."
        ENDMETHOD

        METHOD getShapesCount()
            RETURN SIZE OF shapesCollection
        ENDMETHOD
    ENDCLASS

    CREATE MyCanvas FROM Canvas(1)

    CREATE RedCircle FROM Circle("Red", 10, 20, 5.0)
    CREATE BlueRectangle FROM Rectangle("Blue", 0, 0, 4.0, 6.0)

    CALL MyCanvas.addShape(RedCircle)
    CALL MyCanvas.addShape(BlueRectangle)

    DISPLAY "Total Shapes on Canvas: " + CALL MyCanvas.getShapesCount()
    DISPLAY "Circle Color: " + CALL RedCircle.getColor()
    DISPLAY "Circle Area: " + CALL RedCircle.calculateArea()
    DISPLAY "Rectangle Perimeter: " + CALL BlueRectangle.calculatePerimeter()

END

    CLASS Employee

            name : String

            hireDate : String

        CONSTRUCTOR(empName, id, hDate)
            SET name = empName
            SET employeeID = id
            SET hireDate = hDate

        METHOD getName()

        ENDMETHOD

        METHOD calculatePay()
            RETURN 0.0

    CLASS SalariedEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            fixedMonthlySalary : Double

        CONSTRUCTOR(empName, id, hDate, salary)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET fixedMonthlySalary = salary
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN fixedMonthlySalary
        ENDMETHOD
    ENDCLASS

    CLASS HourlyEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            hoursWorked : Integer
            hourlyRate : Double

        CONSTRUCTOR(empName, id, hDate, hours, rate)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET hoursWorked = hours
            SET hourlyRate = rate
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN hoursWorked * hourlyRate
        ENDMETHOD
    ENDCLASS

    CLASS CommissionEmployee EXTENDS Employee
        PRIVATE ATTRIBUTES:
            basePay : Double
            salesAmount : Double
            commissionRate : Double

        CONSTRUCTOR(empName, id, hDate, bPay, sales, cRate)
            CALL SUPER_CONSTRUCTOR(empName, id, hDate)
            SET basePay = bPay
            SET salesAmount = sales
            SET commissionRate = cRate
        ENDCONSTRUCTOR

        METHOD calculatePay()
            RETURN basePay + (salesAmount * commissionRate)
        ENDMETHOD
    ENDCLASS

    CLASS Payroll
        PRIVATE ATTRIBUTES:
            payrollID : Integer
            employeesList : List of Employee

        CONSTRUCTOR(id)
            SET payrollID = id
            SET employeesList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addEmployee(empObj)
            ADD empObj TO employeesList
            DISPLAY "Employee successfully added to payroll system."
        ENDMETHOD

        METHOD calculateTotalPayout()
            SET totalPayout = 0.0
            SET totalEmployees = SIZE OF employeesList
            
            REPEAT FROM i = 0 TO totalEmployees - 1
                SET totalPayout = totalPayout + CALL employeesList[i].calculatePay()
            ENDREPEAT
            
            RETURN totalPayout
        ENDMETHOD
    ENDCLASS

    CREATE CompanyPayroll FROM Payroll(1)

    CREATE Emp_1 FROM SalariedEmployee("Monther Al-Waili", 7015, "2026-01-01", 1200.00)
    CREATE Emp_2 FROM HourlyEmployee("Ahmed Al-Balushi", 8022, "2026-03-15", 160, 5.00)

    CALL CompanyPayroll.addEmployee(Emp_1)
    CALL CompanyPayroll.addEmployee(Emp_2)

    DISPLAY "Payroll System Live Overview"
    DISPLAY "Employee 1 Name: " + CALL Emp_1.getName()
    DISPLAY "Employee 1 Calculated Pay: OMR " + CALL Emp_1.calculatePay()
    DISPLAY "Employee 2 Calculated Pay: OMR " + CALL Emp_2.calculatePay()
    DISPLAY "Total Corporate Payroll Payout: OMR " + CALL CompanyPayroll.calculateTotalPayout()

END

    CLASS PaymentMethod
        PRIVATE ATTRIBUTES:
            amount : Double
            status : String

        CONSTRUCTOR(amt)
            SET amount = amt
            SET status = "Pending"
        ENDCONSTRUCTOR

        METHOD getAmount()
            RETURN amount
        ENDMETHOD

        METHOD getStatus()
            RETURN status
        ENDMETHOD

        METHOD setStatus(newStatus)
            SET status = newStatus
        ENDMETHOD

        METHOD processPayment()
            RETURN FALSE
        ENDMETHOD
    ENDCLASS

    CLASS CreditCardPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            cardNumber : String
            expiryDate : String

        CONSTRUCTOR(amt, card, expiry)
            CALL SUPER_CONSTRUCTOR(amt)
            SET cardNumber = card
            SET expiryDate = expiry
        ENDCONSTRUCTOR

        METHOD processPayment()
            SET length = CALL cardNumber.length()
            IF length Equals 16 THEN
                CALL setStatus("Completed")
                RETURN TRUE
            ELSE
                CALL setStatus("Failed")
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS BankTransferPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            accountNumber : String
            bankCode : String

        CONSTRUCTOR(amt, acc, code)
            CALL SUPER_CONSTRUCTOR(amt)
            SET accountNumber = acc
            SET bankCode = code
        ENDCONSTRUCTOR

        METHOD processPayment()
            CALL setStatus("Completed")
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CLASS WalletPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            walletID : String
            availableBalance : Double

        CONSTRUCTOR(amt, wallet, balance)
            CALL SUPER_CONSTRUCTOR(amt)
            SET walletID = wallet
            SET availableBalance = balance
        ENDCONSTRUCTOR

        METHOD processPayment()
            SET neededAmount = CALL getAmount()
            IF availableBalance Greater Than Or Equals neededAmount THEN
                CALL setStatus("Completed")
                RETURN TRUE
            ELSE
                CALL setStatus("Failed")
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS OrderItem
        PRIVATE ATTRIBUTES:
            itemName : String
            itemPrice : Double

        CONSTRUCTOR(name, price)
            SET itemName = name
            SET itemPrice = price
        ENDCONSTRUCTOR

        METHOD getItemPrice()
            RETURN itemPrice
        ENDMETHOD
    ENDCLASS

    CLASS Order
        PRIVATE ATTRIBUTES:
            orderID : Integer
            itemsList : List of OrderItem
            paymentObj : PaymentMethod

        CONSTRUCTOR(id)
            SET orderID = id
            SET itemsList = INITIALIZE EMPTY LIST
            SET paymentObj = NULL
        ENDCONSTRUCTOR

        METHOD addItem(itemObj)
            ADD itemObj TO itemsList
        ENDMETHOD

        METHOD calculateTotal()
            SET totalSum = 0.0
            SET totalItems = SIZE OF itemsList
            
            REPEAT FROM i = 0 TO totalItems - 1
                SET totalSum = totalSum + CALL itemsList[i].getItemPrice()
            ENDREPEAT
            
            RETURN totalSum
        ENDMETHOD

        METHOD setPayment(payMethodObj)
            SET paymentObj = payMethodObj
        ENDMETHOD

        METHOD checkout()
            SET success = CALL paymentObj.processPayment()
            IF success Equals TRUE THEN
                DISPLAY "Purchase Completed Successfully!"
            ELSE
                DISPLAY "Purchase Terminated: Payment Declined!"
            ENDIF
        ENDMETHOD
    ENDCLASS

    CREATE StoreOrder FROM Order(88021)
    CREATE Item_1 FROM OrderItem("Gaming Mouse", 25.00)
    CREATE Item_2 FROM OrderItem("Mechanical Keyboard", 45.00)

    CALL StoreOrder.addItem(Item_1)
    CALL StoreOrder.addItem(Item_2)

    SET billTotal = CALL StoreOrder.calculateTotal()
    CREATE MyWalletPay FROM WalletPayment(billTotal, "W-7015", 100.00)
    
    CALL StoreOrder.setPayment(MyWalletPay)
    CALL StoreOrder.checkout()

    DISPLAY "Order Settlement Status: " + CALL MyWalletPay.getStatus()

END
