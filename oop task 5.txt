        PRIVATE ATTRIBUTES:

            SET color = clr
            SET positionX = x
            SET positionY = y
        ENDCONSTRUCTOR

            RETURN color
        ENDMETHOD

        METHOD calculateArea()
            RETURN 0.0
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS Circle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            radius : Double

        CONSTRUCTOR(clr, x, y, r)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET radius = r
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN 3.14 * radius * radius
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * 3.14 * radius
        ENDMETHOD
    ENDCLASS

    CLASS Rectangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            width : Double
            height : Double

        CONSTRUCTOR(clr, x, y, w, h)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET width = w
            SET height = h
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN width * height
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * (width + height)
        ENDMETHOD
    ENDCLASS

    CLASS Triangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            sideA : Double
            sideB : Double
            sideC : Double

        CONSTRUCTOR(clr, x, y, a, b, c)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET sideA = a
            SET sideB = b
            SET sideC = c
        ENDCONSTRUCTOR

        METHOD calculateArea()
            SET s = (sideA + sideB + sideC) / 2.0
            RETURN CALL MATH_SQRT(s * (s - sideA) * (s - sideB) * (s - sideC))
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN sideA + sideB + sideC
        ENDMETHOD
    ENDCLASS

    CLASS Canvas
        PRIVATE ATTRIBUTES:
            canvasID : Integer
            shapesCollection : List of Shape

        CONSTRUCTOR(id)
            SET canvasID = id
            SET shapesCollection = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addShape(shapeObj)
            ADD shapeObj TO shapesCollection
            DISPLAY "Shape successfully drawn on the canvas."
        ENDMETHOD

        METHOD getShapesCount()
            RETURN SIZE OF shapesCollection
        ENDMETHOD
    ENDCLASS

            SET cardNumber = card
            SET expiryDate = expiry

            SET length = CALL cardNumber.length()

            CALL SUPER_CONSTRUCTOR(amt)
            SET walletID = wallet

    CLASS Card
        PRIVATE ATTRIBUTES:
            cardNumber : String
            expiryDate : String
            status : String
            chipObj : Chip

        CONSTRUCTOR(num, expiry, chp)
            SET cardNumber = num
            SET expiryDate = expiry
            SET status = "Active"
            SET chipObj = chp
        ENDCONSTRUCTOR

        METHOD getStatus()
            RETURN status
        ENDMETHOD

        METHOD authorisePurchase(amount)
            RETURN FALSE
        ENDMETHOD
    ENDCLASS

    CLASS DebitCard EXTENDS Card
        PRIVATE ATTRIBUTES:
            availableBalance : Double

        CONSTRUCTOR(num, expiry, chp, balance)
            CALL SUPER_CONSTRUCTOR(num, expiry, chp)
            SET availableBalance = balance
        ENDCONSTRUCTOR

        METHOD authorisePurchase(amount)
            SET currentStatus = CALL getStatus()
            IF currentStatus Equals "Blocked" THEN
                DISPLAY "Authorization Refused: Card is blocked!"
                RETURN FALSE
            ENDIF

            IF availableBalance Greater Than Or Equals amount THEN
                SET availableBalance = availableBalance - amount
                DISPLAY "Purchase Authorized Successfully."
                RETURN TRUE
            ELSE
                DISPLAY "Authorization Refused: Insufficient account balance!"
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS CreditCard EXTENDS Card
        PRIVATE ATTRIBUTES:
            creditLimit : Double
            unpaidPurchasesList : List of Double

        CONSTRUCTOR(num, expiry, chp, limit)
            CALL SUPER_CONSTRUCTOR(num, expiry, chp)
            SET creditLimit = limit
            SET unpaidPurchasesList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addUnpaidPurchase(amount)
            ADD amount TO unpaidPurchasesList
        ENDMETHOD

        METHOD calculateOutstandingBalance()
            SET totalOutstanding = 0.0
            SET totalTransactions = SIZE OF unpaidPurchasesList
            
            REPEAT FROM i = 0 TO totalTransactions - 1
                SET totalOutstanding = totalOutstanding + unpaidPurchasesList[i]
            ENDREPEAT
            
            RETURN totalOutstanding
        ENDMETHOD

        METHOD authorisePurchase(amount)
            SET currentStatus = CALL getStatus()
            IF currentStatus Equals "Blocked" THEN
                DISPLAY "Authorization Refused: Card is blocked!"
                RETURN FALSE
            ENDIF

            SET outstanding = CALL calculateOutstandingBalance()
            IF outstanding + amount Less Than Or Equals creditLimit THEN
                CALL addUnpaidPurchase(amount)
                DISPLAY "Credit Purchase Authorized Successfully."
                RETURN TRUE
            ELSE
                DISPLAY "Authorization Refused: Operation exceeds credit limit!"
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

        PRIVATE ATTRIBUTES:

            listenedSeconds : Double

        CONSTRUCTOR(tlt, auth, pages, cvr, totalSecs, listenedSecs)
            CALL SUPER_CONSTRUCTOR(tlt, auth, pages, cvr)
            SET totalDurationSeconds = totalSecs
            SET listenedSeconds = listenedSecs
        ENDCONSTRUCTOR

        METHOD getProgress()
            IF totalDurationSeconds Equals 0.0 THEN
                RETURN 0.0
            ENDIF
            RETURN (listenedSeconds / totalDurationSeconds) * 100.0
        ENDMETHOD
    ENDCLASS

    CLASS Library
        PRIVATE ATTRIBUTES:
            libraryID : Integer
            publicationsList : List of Publication

        CONSTRUCTOR(id)
            SET libraryID = id
            SET publicationsList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addPublication(pubObj)
            ADD pubObj TO publicationsList
        ENDMETHOD

        METHOD getPublicationsCount()
            RETURN SIZE OF publicationsList
        ENDMETHOD
    ENDCLASS

    CREATE CustomCover FROM CoverImage("/images/cipher_cover.png", "800x1200")
    CREATE Novel_Cipher FROM Novel("The Chronomancer's Cipher", "Al-Monther", 350, CustomCover, 175)
    
    CREATE ReaderLibrary FROM Library(1)
    CALL ReaderLibrary.addPublication(Novel_Cipher)

    DISPLAY "E-book Reader Engine Overview"
    DISPLAY "Total Digital Library Stock: " + CALL ReaderLibrary.getPublicationsCount()
    DISPLAY "Current Publication Title: " + CALL Novel_Cipher.getTitle()
    DISPLAY "Calculated Reading Progress Percentage: " + CALL Novel_Cipher.getProgress() + "%"
