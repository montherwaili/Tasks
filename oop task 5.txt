            SET positionX = x

        METHOD calculateArea()
            RETURN 0.0
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 0.0
        ENDMETHOD

    CLASS Circle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            radius : Double

        CONSTRUCTOR(clr, x, y, r)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET radius = r
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN 3.14 * radius * radius
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * 3.14 * radius
        ENDMETHOD
    ENDCLASS

    CLASS Rectangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            width : Double
            height : Double

        CONSTRUCTOR(clr, x, y, w, h)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET width = w
            SET height = h
        ENDCONSTRUCTOR

        METHOD calculateArea()
            RETURN width * height
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN 2.0 * (width + height)
        ENDMETHOD
    ENDCLASS

    CLASS Triangle EXTENDS Shape
        PRIVATE ATTRIBUTES:
            sideA : Double
            sideB : Double
            sideC : Double

        CONSTRUCTOR(clr, x, y, a, b, c)
            CALL SUPER_CONSTRUCTOR(clr, x, y)
            SET sideA = a
            SET sideB = b
            SET sideC = c
        ENDCONSTRUCTOR

        METHOD calculateArea()
            SET s = (sideA + sideB + sideC) / 2.0
            RETURN CALL MATH_SQRT(s * (s - sideA) * (s - sideB) * (s - sideC))
        ENDMETHOD

        METHOD calculatePerimeter()
            RETURN sideA + sideB + sideC
        ENDMETHOD
    ENDCLASS

    CLASS Canvas
        PRIVATE ATTRIBUTES:
            canvasID : Integer
            shapesCollection : List of Shape

        CONSTRUCTOR(id)
            SET canvasID = id
            SET shapesCollection = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD addShape(shapeObj)
            ADD shapeObj TO shapesCollection
            DISPLAY "Shape successfully drawn on the canvas."
        ENDMETHOD

        METHOD getShapesCount()
            RETURN SIZE OF shapesCollection
        ENDMETHOD
    ENDCLASS

    CLASS PaymentMethod
        PRIVATE ATTRIBUTES:
            amount : Double
            status : String

        CONSTRUCTOR(amt)
            SET this.amount TO amt
            SET this.status TO "Pending"
        ENDCONSTRUCTOR

    CLASS CreditCardPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            cardNumber : String
            expiryDate : String // Formatted as "MM/YY" e.g., "12/30"

        CONSTRUCTOR(amt, card, expiry)
            CALL SUPER_CONSTRUCTOR(amt)
            SET this.cardNumber TO card
            SET this.expiryDate TO expiry
        ENDCONSTRUCTOR

        METHOD processPayment()
            SET cardLength TO SIZE OF this.cardNumber
            SET expiryLength TO SIZE OF this.expiryDate
            
            IF cardLength Equals 16 AND expiryLength Equals 5 THEN
                CALL setStatus("Completed")
                RETURN TRUE
            ELSE
                CALL setStatus("Failed")
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS BankTransferPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            accountNumber : String
            bankCode : String

        CONSTRUCTOR(amt, acc, code)
            CALL SUPER_CONSTRUCTOR(amt)
            SET this.accountNumber TO acc
            SET this.bankCode TO code
        ENDCONSTRUCTOR

        METHOD processPayment()
            SET accLength TO SIZE OF this.accountNumber
            SET codeLength TO SIZE OF this.bankCode
            
            IF accLength Greater Than 8 AND codeLength Equals 4 THEN
                CALL setStatus("Completed")
                RETURN TRUE
            ELSE
                CALL setStatus("Failed")
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS WalletPayment EXTENDS PaymentMethod
        PRIVATE ATTRIBUTES:
            walletID : String
            availableBalance : Double

        CONSTRUCTOR(amt, wallet, balance)
            CALL SUPER_CONSTRUCTOR(amt)
            SET this.walletID TO wallet
            SET this.availableBalance TO balance
        ENDCONSTRUCTOR

        METHOD processPayment()

    CLASS Card

            cardNumber : String
            expiryDate : String
            status : String
            chipObj : Chip

        CONSTRUCTOR(num, expiry, chp)
            SET this.cardNumber TO num
            SET this.expiryDate TO expiry
            SET this.status TO "Active"
            SET this.chipObj TO chp
        ENDCONSTRUCTOR

        METHOD getStatus()
            RETURN this.status
        ENDMETHOD

        ABSTRACT METHOD authorisePurchase(amount)
    ENDCLASS

    CLASS DebitCard EXTENDS Card
        PRIVATE ATTRIBUTES:
            availableBalance : Double

        CONSTRUCTOR(num, expiry, chp, balance)
            CALL SUPER_CONSTRUCTOR(num, expiry, chp)
            SET this.availableBalance TO balance
        ENDCONSTRUCTOR

        METHOD authorisePurchase(amount)
            SET currentStatus TO CALL this.getStatus()
            IF currentStatus Equals "Blocked" THEN
                DISPLAY "Authorization Refused: Card is blocked!"
                RETURN FALSE
            ENDIF

            IF this.availableBalance Greater Than Or Equals amount THEN
                SET this.availableBalance TO this.availableBalance - amount
                DISPLAY "Purchase Authorized Successfully."
                RETURN TRUE
            ELSE
                DISPLAY "Authorization Refused: Insufficient account balance!"
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS CreditCard EXTENDS Card
        PRIVATE ATTRIBUTES:
            creditLimit : Double
            unpaidPurchasesList : List of Double

        CONSTRUCTOR(num, expiry, chp, limit)
            CALL SUPER_CONSTRUCTOR(num, expiry, chp)
            SET this.creditLimit TO limit
            SET this.unpaidPurchasesList TO LIST []
        ENDCONSTRUCTOR

        METHOD addUnpaidPurchase(amount)
            ADD amount TO this.unpaidPurchasesList
        ENDMETHOD

        METHOD calculateOutstandingBalance()
            SET totalOutstanding TO 0.0
            SET totalTransactions TO SIZE OF this.unpaidPurchasesList
            
            REPEAT FROM i = 0 TO totalTransactions - 1
                SET totalOutstanding TO totalOutstanding + this.unpaidPurchasesList[i]
            ENDREPEAT
            
            RETURN totalOutstanding
        ENDMETHOD

        METHOD makePayment(paymentAmount)
            SET outstanding TO CALL this.calculateOutstandingBalance()
            IF paymentAmount Greater Than outstanding THEN
                SET paymentAmount TO outstanding
            ENDIF
            
            SET totalTransactions TO SIZE OF this.unpaidPurchasesList
            IF totalTransactions Greater Than 0 THEN
                SET this.unpaidPurchasesList TO LIST []
                SET newOutstanding TO outstanding - paymentAmount
                IF newOutstanding Greater Than 0.0 THEN
                    ADD newOutstanding TO this.unpaidPurchasesList
                ENDIF
            ENDIF
            DISPLAY "Payment received. Outstanding credit balance updated."
        ENDMETHOD

        METHOD authorisePurchase(amount)
            SET currentStatus TO CALL this.getStatus()
            IF currentStatus Equals "Blocked" THEN
                DISPLAY "Authorization Refused: Card is blocked!"
                RETURN FALSE
            ENDIF

            SET outstanding TO CALL this.calculateOutstandingBalance()
            IF outstanding + amount Less Than Or Equals this.creditLimit THEN
                CALL this.addUnpaidPurchase(amount)
                DISPLAY "Credit Purchase Authorized Successfully."
                RETURN TRUE
            ELSE
                DISPLAY "Authorization Refused: Operation exceeds credit limit!"
                RETURN FALSE
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS StorageUnit
        PRIVATE ATTRIBUTES:
            maxCapacityGB : Double
            usedSpaceGB : Double

        CONSTRUCTOR(maxCap, used)
            SET maxCapacityGB = maxCap
            SET usedSpaceGB = used
        ENDCONSTRUCTOR

        METHOD hasSpace(neededGB)
            IF usedSpaceGB + neededGB Greater Than maxCapacityGB THEN
                RETURN FALSE
            ELSE
                RETURN TRUE
            ENDIF
        ENDMETHOD

        METHOD consumeSpace(neededGB)
            SET usedSpaceGB = usedSpaceGB + neededGB
        ENDMETHOD
    ENDCLASS

    CLASS Device
        PRIVATE ATTRIBUTES:
            name : String
            roomLocation : String
            isOn : Boolean

        CONSTRUCTOR(dName, location)
            SET name = dName
            SET roomLocation = location
            SET isOn = FALSE
        ENDCONSTRUCTOR

        METHOD turnOn()
            SET isOn = TRUE
        ENDMETHOD

        METHOD getIsOn()
            RETURN isOn
        ENDMETHOD

        METHOD getPowerUsage()
            RETURN 0.0
        ENDMETHOD
    ENDCLASS

    CLASS SmartLight EXTENDS Device
        PRIVATE ATTRIBUTES:
            brightnessLevel : Integer

        CONSTRUCTOR(dName, location, brightness)
            CALL SUPER_CONSTRUCTOR(dName, location)
            SET brightnessLevel = brightness
        ENDCONSTRUCTOR

        METHOD getPowerUsage()
            IF CALL getIsOn() Equals FALSE THEN
                RETURN 0.0
            ENDIF
            RETURN brightnessLevel * 0.15
        ENDMETHOD
    ENDCLASS

    CLASS SmartThermostat EXTENDS Device
        PRIVATE ATTRIBUTES:
            targetTemperature : Double
            coolingIntensity : Double

        CONSTRUCTOR(dName, location, target, intensity)
            CALL SUPER_CONSTRUCTOR(dName, location)
            SET targetTemperature = target
            SET coolingIntensity = intensity
        ENDCONSTRUCTOR

        METHOD getPowerUsage()
            IF CALL getIsOn() Equals FALSE THEN
                RETURN 0.0
            ENDIF
            RETURN coolingIntensity * 12.0
        ENDMETHOD
    ENDCLASS

    CLASS SmartCamera EXTENDS Device
        PRIVATE ATTRIBUTES:
            isRecording : Boolean
            storageObj : StorageUnit

        CONSTRUCTOR(dName, location, strg)
            CALL SUPER_CONSTRUCTOR(dName, location)
            SET isRecording = FALSE
            SET storageObj = strg
        ENDCONSTRUCTOR

        METHOD startRecording()
            IF CALL storageObj.hasSpace(5.0) Equals FALSE THEN
                DISPLAY "Recording Terminated: Storage unit is full!"
                SET isRecording = FALSE
                RETURN FALSE
            ENDIF
            SET isRecording = TRUE
            CALL storageObj.consumeSpace(5.0)
            DISPLAY "Camera successfully started recording."
            RETURN TRUE
        ENDMETHOD

        METHOD getPowerUsage()
            IF CALL getIsOn() Equals FALSE THEN
                RETURN 0.0
            ENDIF
            IF isRecording Equals TRUE THEN
                RETURN 25.0
            ELSE
                RETURN 5.0
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS Room
        PRIVATE ATTRIBUTES:
            roomName : String
            devicesList : List of Device

        CONSTRUCTOR(name)
            SET roomName = name
            SET devicesList = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD installDevice(deviceObj)
            ADD deviceObj TO devicesList
        ENDMETHOD

        METHOD calculateTotalPowerUsage()
            SET totalPower = 0.0
            SET totalDevices = SIZE OF devicesList
            
            REPEAT FROM i = 0 TO totalDevices - 1
                SET totalPower = totalPower + CALL devicesList[i].getPowerUsage()
            ENDREPEAT
            
            RETURN totalPower
        ENDMETHOD
    ENDCLASS

    CALL NestThermostat.turnOn()
    CALL SecurityCam.turnOn()
    
    CALL SecurityCam.startRecording() // Will consume 5GB but has only 2GB left -> Should fail

    DISPLAY "Smart Home Command Dashboard"
    DISPLAY "Total Real-time Room Power Consumption: " + CALL LivingRoom.calculateTotalPowerUsage() + " Watts"

END
