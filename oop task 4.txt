    CLASS Question
        PRIVATE ATTRIBUTES:
            questionID : Integer
            text : String
            correctAnswer : String
            pointValue : Integer
            optionsList : List of String

        CONSTRUCTOR(id, qText, correctAns, points)
            SET this.questionID TO id
            SET this.text TO qText
            SET this.correctAnswer TO correctAns
            SET this.pointValue TO points
            SET this.optionsList TO LIST []
        ENDCONSTRUCTOR

        METHOD getQuestionID()
            RETURN this.questionID
        ENDMETHOD

        METHOD getCorrectAnswer()
            RETURN this.correctAnswer
        ENDMETHOD

        METHOD getPointValue()
            RETURN this.pointValue
        ENDMETHOD

        METHOD addOption(optionText)
            ADD optionText TO this.optionsList
        ENDMETHOD

        METHOD isValidOption(studentAnswer)
            SET totalOptions TO SIZE OF this.optionsList
            REPEAT FROM i = 0 TO totalOptions - 1
                IF this.optionsList[i] Equals studentAnswer THEN
                    RETURN TRUE
                ENDIF
            ENDREPEAT
            RETURN FALSE
        ENDMETHOD
    ENDCLASS

    CLASS Exam

    CLASS StudentAnswer
        PRIVATE ATTRIBUTES:
            associatedQuestionID : Integer
            selectedOption : String

        CONSTRUCTOR(qID, option)
            SET this.associatedQuestionID TO qID
            SET this.selectedOption TO option
        ENDCONSTRUCTOR

        METHOD getAssociatedQuestionID()
            RETURN this.associatedQuestionID
        ENDMETHOD

        METHOD getSelectedOption()
            RETURN this.selectedOption
        ENDMETHOD
    ENDCLASS

    CLASS ExamAttempt
        PRIVATE ATTRIBUTES:
            attemptID : Integer
            examInfo : Exam
            studentAnswersList : List of StudentAnswer
            elapsedTimeInMinutes : Integer

        CONSTRUCTOR(id, ex)
            SET this.attemptID TO id
            SET this.examInfo TO ex
            SET this.elapsedTimeInMinutes TO 0
            SET this.studentAnswersList TO LIST []
        ENDCONSTRUCTOR

        METHOD setElapsedTime(minutes)
            SET this.elapsedTimeInMinutes TO minutes
        ENDMETHOD

        METHOD selectAnswer(questionObj, optionText)
            SET maxLimit TO CALL this.examInfo.getTimeLimit()
            IF this.elapsedTimeInMinutes Greater Than maxLimit THEN
                DISPLAY "Transaction Terminated: Time limit has passed! Cannot accept more answers."
                RETURN FALSE
            ENDIF

            IF CALL questionObj.isValidOption(optionText) Equals FALSE THEN
                DISPLAY "Transaction Terminated: Invalid option selection for this question!"
                RETURN FALSE
            ENDIF

            CREATE newAnswer FROM StudentAnswer(CALL questionObj.getQuestionID(), optionText)
            ADD newAnswer TO this.studentAnswersList
            RETURN TRUE
        ENDMETHOD

        METHOD calculateScore()
            SET totalScore TO 0
            SET totalStudentAnswers TO SIZE OF this.studentAnswersList
            SET examQuestions TO CALL this.examInfo.getQuestionsList()
            SET totalExamQuestions TO SIZE OF examQuestions

            REPEAT FROM i = 0 TO totalExamQuestions - 1
                SET currentQuestion TO examQuestions[i]
                SET currentQuestionID TO CALL currentQuestion.getQuestionID()
                SET isAnswerFound TO FALSE
                SET studentSelectedOption TO ""

                REPEAT FROM j = 0 TO totalStudentAnswers - 1
                    IF CALL this.studentAnswersList[j].getAssociatedQuestionID() Equals currentQuestionID THEN
                        SET isAnswerFound TO TRUE
                        SET studentSelectedOption TO CALL this.studentAnswersList[j].getSelectedOption()
                    ENDIF
                ENDREPEAT

                IF isAnswerFound Equals TRUE THEN
                    IF studentSelectedOption Equals CALL currentQuestion.getCorrectAnswer() THEN
                        SET totalScore TO totalScore + CALL currentQuestion.getPointValue()
                    ENDIF
                ENDIF
            ENDREPEAT

            RETURN totalScore
        ENDMETHOD
    ENDCLASS

    CLASS Student
        PRIVATE ATTRIBUTES:
            studentID : Integer
            name : String
            allAttemptsHistory : List of ExamAttempt

        CONSTRUCTOR(id, sName)
            SET this.studentID TO id
            SET this.name TO sName
            SET this.allAttemptsHistory TO LIST []
        ENDCONSTRUCTOR

        METHOD logAttempt(attemptObj)
            ADD attemptObj TO this.allAttemptsHistory
            DISPLAY "Exam attempt successfully locked and archived."
        ENDMETHOD
    ENDCLASS

    CREATE MidtermExam FROM Exam(501, "Java OOP Exam", 30)

    CALL Q1.addOption("B")
    
    CREATE Q2 FROM Question(2, "What does git status do?", "B", 5)
    CALL Q2.addOption("A")
    CALL Q2.addOption("B")
    
    CALL MidtermExam.addQuestion(Q1)
    CALL MidtermExam.addQuestion(Q2)

    CREATE Std_Monther FROM Student(7015, "Monther Al-Waili")

    CREATE Attempt_1 FROM ExamAttempt(1, MidtermExam)
    CALL Attempt_1.setElapsedTime(15)
    CALL Attempt_1.selectAnswer(Q1, "A")
    CALL Attempt_1.selectAnswer(Q2, "B")
    CALL Std_Monther.logAttempt(Attempt_1)

    DISPLAY "First Verified Attempt Score Results: " + CALL Attempt_1.calculateScore() + " Points"

END

    CLASS Booking
        PRIVATE ATTRIBUTES:
            bookingID : Integer

            startTime : Integer 
            endTime : Integer  
            durationHours : Integer

        CONSTRUCTOR(id, rm, start, end, duration)
            SET bookingID = id
            SET roomObj = rm
            SET startTime = start
            SET endTime = end
            SET durationHours = duration
        ENDCONSTRUCTOR

        METHOD getDurationHours()
            RETURN durationHours
        ENDMETHOD

        METHOD getStartTime()
            RETURN startTime
        ENDMETHOD

        METHOD getEndTime()
            RETURN endTime
        ENDMETHOD

        METHOD getRoomObj()
            RETURN roomObj
        ENDMETHOD
    ENDCLASS

    CLASS Member
        PRIVATE ATTRIBUTES:
            memberID : Integer
            name : String
            planDetails : MembershipPlan
            activeBookings : List of Booking

        CONSTRUCTOR(id, mName, plan)
            SET memberID = id
            SET name = mName
            SET planDetails = plan
            SET activeBookings = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD calculateUsedHours()
            SET totalUsed = 0
            SET totalBookings = SIZE OF activeBookings
            
            REPEAT FROM i = 0 TO totalBookings - 1
                SET totalUsed = totalUsed + CALL activeBookings[i].getDurationHours()
            ENDREPEAT
            
            RETURN totalUsed
        ENDMETHOD

        METHOD hasTimeOverlap(newBooking)
            SET totalBookings = SIZE OF activeBookings
            SET targetRoomID = CALL CALL newBooking.getRoomObj().getRoomID()
            SET newStart = CALL newBooking.getStartTime()
            SET newEnd = CALL newBooking.getEndTime()

            REPEAT FROM i = 0 TO totalBookings - 1
                SET existingBooking = activeBookings[i]
                SET existingRoomID = CALL CALL existingBooking.getRoomObj().getRoomID()
                
                IF existingRoomID == targetRoomID THEN
                    SET extStart = CALL existingBooking.getStartTime()
                    SET extEnd = CALL existingBooking.getEndTime()
                    
                    IF newStart < extEnd AND newEnd > extStart THEN
                        RETURN TRUE
                    ENDIF
                ENDIF
            ENDREPEAT
            RETURN FALSE
        ENDMETHOD

        METHOD requestBooking(bookingObj)
            SET usedHours = CALL calculateUsedHours()
            SET quotaLimit = CALL planDetails.getMonthlyQuota()
            SET requestedHours = CALL bookingObj.getDurationHours()

            IF usedHours + requestedHours > quotaLimit THEN
                DISPLAY "Booking Refused: Member has exceeded their monthly plan quota hours!"
                RETURN FALSE
            ENDIF

            IF CALL hasTimeOverlap(bookingObj) == TRUE THEN
                DISPLAY "Booking Refused: Time slot overlap detected for this meeting room!"
                RETURN FALSE
            ENDIF

            ADD bookingObj TO activeBookings
            DISPLAY "Booking successfully confirmed and registered."
            RETURN TRUE
        ENDMETHOD
    ENDCLASS

    CREATE ExecutivePlan FROM MembershipPlan("Premium Executive", 5)
    CREATE MeetingRoom_A FROM Room(201, "Conference Room A")
    CREATE Member_Monther FROM Member(7015, "Monther Al-Waili", ExecutivePlan)

    CREATE BOOK_1 FROM Booking(1, MeetingRoom_A, 1000, 1300, 3)
    CREATE BOOK_2 FROM Booking(2, MeetingRoom_A, 1100, 1400, 3) 

    CALL Member_Monther.requestBooking(BOOK_1)
    CALL Member_Monther.requestBooking(BOOK_2) 

    DISPLAY "Live Calculated Remaining Hours: " + (CALL ExecutivePlan.getMonthlyQuota() - CALL Member_Monther.calculateUsedHours())

END

    CLASS Trainer
        PRIVATE ATTRIBUTES:
            trainerID : Integer
            name : String
            masterSchedule : List of Session

        CONSTRUCTOR(id, tName)
            SET trainerID = id
            SET name = tName
            SET masterSchedule = INITIALIZE EMPTY LIST
        ENDCONSTRUCTOR

        METHOD hasTrainerConflict(newSession)
            SET totalSessions = SIZE OF masterSchedule
            SET newDate = CALL newSession.getDate()
            SET newStart = CALL newSession.getStartTime()
            SET newDuration = CALL newSession.getDurationMinutes()
            SET newEnd = newStart + (newDuration / 60 * 100) 
            REPEAT FROM i = 0 TO totalSessions - 1
                SET existingSession = masterSchedule[i]
                IF CALL existingSession.getDate() == newDate THEN
                    SET extStart = CALL existingSession.getStartTime()
                    SET extDuration = CALL existingSession.getDurationMinutes()
                    SET extEnd = extStart + (extDuration / 60 * 100)
                    
                    IF newStart < extEnd AND newEnd > extStart THEN
                        RETURN TRUE
                    ENDIF
                ENDIF
            ENDREPEAT
            RETURN FALSE
        ENDMETHOD

        METHOD scheduleSession(sessionObj, clientObj)
            IF CALL hasTrainerConflict(sessionObj) == TRUE THEN
                DISPLAY "Scheduling Terminated: Trainer is already booked with another client at this time!"
                RETURN FALSE
            ENDIF

            ADD sessionObj TO masterSchedule
            CALL clientObj.addSessionToClient(sessionObj)
            DISPLAY "Fitness session successfully scheduled on trainer master schedule."
            RETURN TRUE
        ENDMETHOD

        METHOD getTotalScheduledSessions()
            RETURN SIZE OF masterSchedule
        ENDMETHOD
    ENDCLASS

    CREATE FitnessTrainer FROM Trainer(1, "Captain Salem")
    CREATE Cust_Monther FROM Client(7015, "Monther Al-Waili", "+96899998888")

    CREATE SESSION_1 FROM Session(101, "2026-07-15", 1600, 60, "Cardio & Crossfit", Cust_Monther)
    CREATE SESSION_2 FROM Session(102, "2026-07-15", 1630, 60, "Weightlifting", Cust_Monther) 

    CALL FitnessTrainer.scheduleSession(SESSION_1, Cust_Monther)
    CALL FitnessTrainer.scheduleSession(SESSION_2, Cust_Monther)

    DISPLAY "Trainer Master Total Active Booked Sessions: " + CALL FitnessTrainer.getTotalScheduledSessions()

END
