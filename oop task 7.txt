BEGIN

    CLASS DocumentationFolder
        PRIVATE ATTRIBUTES:
            folderID : Integer
            documentsList : List of String

        CONSTRUCTOR(id)
            SET this.folderID TO id
            SET this.documentsList TO LIST []
        ENDCONSTRUCTOR

        METHOD addDocument(documentName)
            ADD documentName TO this.documentsList
        ENDMETHOD

        METHOD getDocumentsCount()
            RETURN SIZE OF this.documentsList
        ENDMETHOD
    ENDCLASS

    CLASS Claim
        PRIVATE ATTRIBUTES:
            claimNumber : String
            dateFiled : String
            statusStage : Integer 
            docsFolder : DocumentationFolder

        CONSTRUCTOR(num, cDate, folderObj)
            SET this.claimNumber TO num
            SET this.dateFiled TO cDate
            SET this.statusStage TO 1
            SET this.docsFolder TO folderObj
        ENDCONSTRUCTOR

        METHOD getStatusStage()
            RETURN this.statusStage
        ENDMETHOD

        METHOD advanceStatus(nextStage)
            IF nextStage Equals this.statusStage + 1 THEN
                SET this.statusStage TO nextStage
                RETURN TRUE
            ELSE
                DISPLAY "Transaction Denied: Invalid status sequence progression!"
                RETURN FALSE
            ENDIF
        ENDMETHOD

        ABSTRACT METHOD calculatePayout()
    ENDCLASS

    CLASS VehicleClaim EXTENDS Claim
        PRIVATE ATTRIBUTES:
            repairEstimate : Double
            fixedExcess : Double

        CONSTRUCTOR(num, cDate, folderObj, estimate, excess)
            CALL SUPER_CONSTRUCTOR(num, cDate, folderObj)
            SET this.repairEstimate TO estimate
            SET this.fixedExcess TO excess
        ENDCONSTRUCTOR

        METHOD calculatePayout()
            IF this.repairEstimate Less Than Or Equals this.fixedExcess THEN
                RETURN 0.0
            ENDIF
            RETURN this.repairEstimate - this.fixedExcess
        ENDMETHOD
    ENDCLASS

    CLASS MedicalClaim EXTENDS Claim
        PRIVATE ATTRIBUTES:
            treatmentCosts : Double
            medicalCoverageLimit : Double

        CONSTRUCTOR(num, cDate, folderObj, costs, limit)
            CALL SUPER_CONSTRUCTOR(num, cDate, folderObj)
            SET this.treatmentCosts TO costs
            SET this.medicalCoverageLimit TO limit
        ENDCONSTRUCTOR

        METHOD calculatePayout()
            IF this.treatmentCosts Greater Than this.medicalCoverageLimit THEN
                RETURN this.medicalCoverageLimit
            ELSE
                RETURN this.treatmentCosts
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS PropertyClaim EXTENDS Claim
        PRIVATE ATTRIBUTES:
            estimatedDamageValue : Double
            isProperlyMaintained : Boolean

        CONSTRUCTOR(num, cDate, folderObj, damage, maintenanceFlag)
            CALL SUPER_CONSTRUCTOR(num, cDate, folderObj)
            SET this.estimatedDamageValue TO damage
            SET this.isProperlyMaintained TO maintenanceFlag
        ENDCONSTRUCTOR

        METHOD calculatePayout()
            IF this.isProperlyMaintained Equals TRUE THEN
                RETURN this.estimatedDamageValue
            ELSE
                RETURN this.estimatedDamageValue - (this.estimatedDamageValue * 0.20)
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS Policy
        PRIVATE ATTRIBUTES:
            policyNumber : String
            claimsHistory : List of Claim

        CONSTRUCTOR(num)
            SET this.policyNumber TO num
            SET this.claimsHistory TO LIST []
        ENDCONSTRUCTOR

        METHOD fileClaim(claimObj)
            ADD claimObj TO this.claimsHistory
            DISPLAY "Insurance claim successfully registered under policy."
        ENDMETHOD

        METHOD calculateTotalPolicyPayout()
            SET totalPaid TO 0.0
            SET totalClaims TO SIZE OF this.claimsHistory
            
            REPEAT FROM i = 0 TO totalClaims - 1
                IF CALL this.claimsHistory[i].getStatusStage() Greater Than Or Equals 3 THEN
                    SET totalPaid TO totalPaid + CALL this.claimsHistory[i].calculatePayout()
                ENDIF
            ENDREPEAT
            
            RETURN totalPaid
        ENDMETHOD
    ENDCLASS

    CREATE ClientFolder FROM DocumentationFolder(551)
    CALL ClientFolder.addDocument("Car_Accident_Photo.jpg")
    CALL ClientFolder.addDocument("Garage_Repair_Invoice.pdf")

    CREATE MontherPolicy FROM Policy("POL-9942")
    CREATE AccidentClaim FROM VehicleClaim("CLM-101", "2026-07-14", ClientFolder, 1500.00, 200.00)

    CALL MontherPolicy.fileClaim(AccidentClaim)
    CALL AccidentClaim.advanceStatus(2)
    CALL AccidentClaim.advanceStatus(3)

    DISPLAY "Corporate Claims Processing Dashboard"
    DISPLAY "Verified Dynamic Claim Payout: OMR " + CALL AccidentClaim.calculatePayout()
    DISPLAY "Total Active Policy Disbursed Capital: OMR " + CALL MontherPolicy.calculateTotalPolicyPayout()

END
STOP

BEGIN

    CLASS ConditionReport
        PRIVATE ATTRIBUTES:
            reportID : Integer
            damageNotes : String

        CONSTRUCTOR(id, notes)
            SET this.reportID TO id
            SET this.damageNotes TO notes
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS LoanItem
        PRIVATE ATTRIBUTES:
            title : String
            barcode : String
            loanDurationDays : Integer
            itemCondition : ConditionReport

        CONSTRUCTOR(tlt, code, duration, reportObj)
            SET this.title TO tlt
            SET this.barcode TO code
            SET this.loanDurationDays TO duration
            SET this.itemCondition TO reportObj
        ENDCONSTRUCTOR

        ABSTRACT METHOD calculateLateFee(daysOverdue)
    ENDCLASS

    CLASS StandardBook EXTENDS LoanItem
        PRIVATE ATTRIBUTES:
            flatFeePerDay : Double

        CONSTRUCTOR(tlt, code, duration, reportObj, fee)
            CALL SUPER_CONSTRUCTOR(tlt, code, duration, reportObj)
            SET this.flatFeePerDay TO fee
        ENDCONSTRUCTOR

        METHOD calculateLateFee(daysOverdue)
            IF daysOverdue Less Than Or Equals 0 THEN
                RETURN 0.0
            ENDIF
            RETURN daysOverdue * this.flatFeePerDay
        ENDMETHOD
    ENDCLASS

    CLASS ReferenceTextbook EXTENDS LoanItem
        PRIVATE ATTRIBUTES:
            premiumFeePerDay : Double

        CONSTRUCTOR(tlt, code, duration, reportObj, premiumFee)
            CALL SUPER_CONSTRUCTOR(tlt, code, duration, reportObj)
            SET this.premiumFeePerDay TO premiumFee
        ENDCONSTRUCTOR

        METHOD getPremiumFeePerDay()
            RETURN this.premiumFeePerDay
        ENDMETHOD

        METHOD calculateLateFee(daysOverdue)
            IF daysOverdue Less Than Or Equals 0 THEN
                RETURN 0.0
            ENDIF
            RETURN daysOverdue * this.premiumFeePerDay
        ENDMETHOD
    ENDCLASS

    CLASS JournalIssue EXTENDS ReferenceTextbook
        PRIVATE ATTRIBUTES:
            isRenewable : Boolean

        CONSTRUCTOR(tlt, code, duration, reportObj, premiumFee)
            CALL SUPER_CONSTRUCTOR(tlt, code, duration, reportObj, premiumFee)
            SET this.isRenewable TO TRUE
        ENDCONSTRUCTOR

        METHOD calculateLateFee(daysOverdue)
            IF daysOverdue Greater Than 0 THEN
                SET this.isRenewable TO FALSE
            ENDIF
            RETURN CALL SUPER_METHOD_calculateLateFee(daysOverdue)
        ENDMETHOD

        METHOD getIsRenewable()
            RETURN this.isRenewable
        ENDMETHOD
    ENDCLASS

    CLASS StudentAccount
        PRIVATE ATTRIBUTES:
            studentID : Integer
            currentLoansList : List of LoanItem

        CONSTRUCTOR(id)
            SET this.studentID TO id
            SET this.currentLoansList TO LIST []
        ENDCONSTRUCTOR

        METHOD borrowItem(loanItemObj)
            ADD loanItemObj TO this.currentLoansList
            DISPLAY "Material successfully checked out and added to student account."
        ENDMETHOD

        METHOD calculateTotalAccountFines(daysOverdue)
            SET totalFines TO 0.0
            SET totalItems TO SIZE OF this.currentLoansList
            
            REPEAT FROM i = 0 TO totalItems - 1
                SET totalFines TO totalFines + CALL this.currentLoansList[i].calculateLateFee(daysOverdue)
            ENDREPEAT
            
            RETURN totalFines
        ENDMETHOD
    ENDCLASS

    CREATE InitialReport FROM ConditionReport(1, "Slight wear on spine, pages clean.")
    CREATE AcademicJournal FROM JournalIssue("International Journal of OOP", "J-9942", 1, InitialReport, 5.00)
    
    CREATE MontherAccount FROM StudentAccount(7015)
    CALL MontherAccount.borrowItem(AcademicJournal)

    SET fineTotal TO CALL AcademicJournal.calculateLateFee(3)

    DISPLAY "University Library Academic Loan Station"
    DISPLAY "Journal Calculated Fine for 3 Days Late: OMR " + fineTotal
    DISPLAY "Is Journal Material Renewable Now? : " + CALL AcademicJournal.getIsRenewable()
    DISPLAY "Total Active Student Outstanding Fines: OMR " + CALL MontherAccount.calculateTotalAccountFines(3)

END
