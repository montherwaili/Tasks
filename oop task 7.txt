CLASS Claim
    PRIVATE ATTRIBUTES:
        claimNumber : String
        dateFiled : String
        statusStage : Integer 
        docsFolder : DocumentationFolder

    CONSTRUCTOR(num, cDate, folderObj)
        SET this.claimNumber TO num
        SET this.dateFiled TO cDate
        SET this.statusStage TO 1
        SET this.docsFolder TO folderObj
    ENDCONSTRUCTOR

    METHOD getStatusStage()
        RETURN this.statusStage
    ENDMETHOD

    METHOD advanceStatus(nextStage)
        IF nextStage Equals this.statusStage + 1 THEN
            SET this.statusStage TO nextStage
            RETURN TRUE
        ELSE
            DISPLAY "Transaction Denied: Invalid status sequence progression!"
            RETURN FALSE
        ENDIF
    ENDMETHOD

    ABSTRACT METHOD calculatePayout()
ENDCLASS

CLASS VehicleClaim INHERITS Claim
    PRIVATE ATTRIBUTES:
        repairEstimate : Double
        fixedExcess : Double

    CONSTRUCTOR(num, cDate, folderObj, estimate, excess)
        CALL SUPER(num, cDate, folderObj)
        SET this.repairEstimate TO estimate
        SET this.fixedExcess TO excess
    ENDCONSTRUCTOR

    METHOD calculatePayout()
        IF this.repairEstimate Less Than Or Equals this.fixedExcess THEN
            RETURN 0.0
        ENDIF
        RETURN this.repairEstimate - this.fixedExcess
    ENDMETHOD
ENDCLASS

CLASS MedicalClaim INHERITS Claim
    PRIVATE ATTRIBUTES:
        treatmentCosts : Double
        medicalCoverageLimit : Double

    CONSTRUCTOR(num, cDate, folderObj, costs, limit)
        CALL SUPER(num, cDate, folderObj)
        SET this.treatmentCosts TO costs
        SET this.medicalCoverageLimit TO limit
    ENDCONSTRUCTOR

    METHOD calculatePayout()
        IF this.treatmentCosts Greater Than this.medicalCoverageLimit THEN
            RETURN this.medicalCoverageLimit
        ELSE
            RETURN this.treatmentCosts
        ENDIF
    ENDMETHOD
ENDCLASS

CLASS PropertyClaim INHERITS Claim
    PRIVATE ATTRIBUTES:
        estimatedDamageValue : Double
        isProperlyMaintained : Boolean

    CONSTRUCTOR(num, cDate, folderObj, damage, maintenanceFlag)
        CALL SUPER(num, cDate, folderObj)
        SET this.estimatedDamageValue TO damage
        SET this.isProperlyMaintained TO maintenanceFlag
    ENDCONSTRUCTOR

    METHOD calculatePayout()
        IF this.isProperlyMaintained Equals TRUE THEN
            RETURN this.estimatedDamageValue
        ELSE
            RETURN this.estimatedDamageValue - (this.estimatedDamageValue * 0.20)
        ENDIF
    ENDMETHOD
ENDCLASS

CLASS Policy
    PRIVATE ATTRIBUTES:
        policyNumber : String
        claimsHistory : LIST<Claim>

    CONSTRUCTOR(num)
        SET this.policyNumber TO num
        SET this.claimsHistory TO LIST []
    ENDCONSTRUCTOR

    METHOD fileClaim(claimObj)
        ADD claimObj TO this.claimsHistory
        DISPLAY "Insurance claim successfully registered under policy."
    ENDMETHOD

    METHOD calculateTotalPolicyPayout()
        SET totalPaid TO 0.0
        SET totalClaims TO SIZE OF this.claimsHistory
        
        REPEAT FROM i = 0 TO totalClaims - 1
            SET currentClaim TO this.claimsHistory[i]
            IF CALL currentClaim.getStatusStage() Greater Than Or Equals 3 THEN
                SET totalPaid TO totalPaid + CALL currentClaim.calculatePayout()
            ENDIF
        ENDREPEAT
        
        RETURN totalPaid
    ENDMETHOD
ENDCLASS

CLASS WaterRide INHERITS Ride
    PRIVATE ATTRIBUTES:
        fitsPerBoat : Integer
        rideCycleTimeMinutes : Integer
        safetyLoadingDelayMinutes : Integer

    CONSTRUCTOR(name, loc, logObj, fits, cycle, delay)
        CALL SUPER(name, loc, logObj)
        SET this.fitsPerBoat TO fits
        SET this.rideCycleTimeMinutes TO cycle
        SET this.safetyLoadingDelayMinutes TO delay
    ENDCONSTRUCTOR

    METHOD calculateWaitTime(peopleInLineCount)
        IF this.fitsPerBoat Equals 0 THEN
            RETURN 0.0
        ENDIF
        SET totalBoatsNeeded TO CALL MATH_CEIL(peopleInLineCount / this.fitsPerBoat)
        RETURN (totalBoatsNeeded * this.rideCycleTimeMinutes) + this.safetyLoadingDelayMinutes
    ENDMETHOD
ENDCLASS

CLASS HourlyInvoice INHERITS Invoice
    PRIVATE ATTRIBUTES:
        loggedHours : Integer
        hourlyRate : Double
        isApprovedByClient : Boolean

    CONSTRUCTOR(num, issued, client, terms, hours, rate, approvedFlag)
        CALL SUPER(num, issued, client, terms)
        SET this.loggedHours TO hours
        SET this.hourlyRate TO rate
        SET this.isApprovedByClient TO approvedFlag
    ENDCONSTRUCTOR

    METHOD calculateTotalAmountDue()
        IF this.isApprovedByClient Equals FALSE THEN
            RETURN 0.0
        ENDIF
        SET baseAmount TO this.loggedHours * this.hourlyRate
        SET interestRate TO CALL CALL this.getTermsObj().getLatePaymentInterestRate()
        RETURN baseAmount + (baseAmount * interestRate)
    ENDMETHOD
ENDCLASS

CLASS FixedPriceInvoice INHERITS Invoice
    PRIVATE ATTRIBUTES:
        agreedProjectFee : Double

    CONSTRUCTOR(num, issued, client, terms, fee)
        CALL SUPER(num, issued, client, terms)
        SET this.agreedProjectFee TO fee
    ENDCONSTRUCTOR

    METHOD getAgreedProjectFee()
        RETURN this.agreedProjectFee
    ENDMETHOD

    METHOD calculateTotalAmountDue()
        SET interestRate TO CALL CALL this.getTermsObj().getLatePaymentInterestRate()
        RETURN this.agreedProjectFee + (this.agreedProjectFee * interestRate)
    ENDMETHOD
ENDCLASS

CLASS RetainerInvoice INHERITS FixedPriceInvoice
    PRIVATE ATTRIBUTES:
        includedHourAllowance : Integer
        actualHoursWorked : Integer
        extraHourlyRate : Double

    CONSTRUCTOR(num, issued, client, terms, fee, allowance, actual, extraRate)
        CALL SUPER(num, issued, client, terms, fee)
        SET this.includedHourAllowance TO allowance
        SET this.actualHoursWorked TO actual
        SET this.extraHourlyRate TO extraRate
    ENDCONSTRUCTOR

    METHOD calculateTotalAmountDue()
        SET baseFee TO CALL SUPER_METHOD_calculateTotalAmountDue()
        IF this.actualHoursWorked Less Than Or Equals this.includedHourAllowance THEN
            RETURN baseFee
        ELSE
            SET extraHours TO this.actualHoursWorked - this.includedHourAllowance
            SET extraCharges TO extraHours * this.extraHourlyRate
            SET interestRate TO CALL CALL this.getTermsObj().getLatePaymentInterestRate()
            RETURN baseFee + extraCharges + (extraCharges * interestRate)
        ENDIF
    ENDMETHOD
ENDCLASS

START
BEGIN

    CLASS HardwareSpecification
        PRIVATE ATTRIBUTES:
            cpuCores : Integer
            ramGB : Integer
            storageTB : Double

        CONSTRUCTOR(cpu, ram, storage)
            SET this.cpuCores TO cpu
            SET this.ramGB TO ram
            SET this.storageTB TO storage
        ENDCONSTRUCTOR
    ENDCLASS

    CLASS Server
        PRIVATE ATTRIBUTES:
            serialNumber : String
            uptimeSeconds : Integer
            hardwareSpecs : HardwareSpecification

        CONSTRUCTOR(serial, uptime, specs)
            SET this.serialNumber TO serial
            SET this.uptimeSeconds TO uptime
            SET this.hardwareSpecs TO specs
        ENDCONSTRUCTOR

        ABSTRACT METHOD calculateLoadPercentage()
    ENDCLASS

    CLASS WebServer EXTENDS Server
        PRIVATE ATTRIBUTES:
            activeConnections : Integer
            maxConnectionLimit : Integer

        CONSTRUCTOR(serial, uptime, specs, active, maxLimit)
            CALL SUPER_CONSTRUCTOR(serial, uptime, specs)
            SET this.activeConnections TO active
            SET this.maxConnectionLimit TO maxLimit
        ENDCONSTRUCTOR

        METHOD calculateLoadPercentage()
            IF maxConnectionLimit Equals 0 THEN
                RETURN 0.0
            ENDIF
            RETURN (activeConnections / maxConnectionLimit) * 100.0
        ENDMETHOD
    ENDCLASS

    CLASS DatabaseServer EXTENDS Server
        PRIVATE ATTRIBUTES:
            queryThroughput : Double
            maxQPSCapacity : Double

        CONSTRUCTOR(serial, uptime, specs, throughput, qps)
            CALL SUPER_CONSTRUCTOR(serial, uptime, specs)
            SET this.queryThroughput TO throughput
            SET this.maxQPSCapacity TO qps
        ENDCONSTRUCTOR

        METHOD getQueryThroughput()
            RETURN this.queryThroughput
        ENDMETHOD

        METHOD getMaxQPSCapacity()
            RETURN this.maxQPSCapacity
        ENDMETHOD

        METHOD calculateLoadPercentage()
            IF maxQPSCapacity Equals 0.0 THEN
                RETURN 0.0
            ENDIF
            SET calculatedLoad TO (queryThroughput / maxQPSCapacity) * 100.0
            IF calculatedLoad Greater Than Or Equals 100.0 THEN
                DISPLAY "Security Warning: Server at full capacity! Connections locked."
                RETURN 100.0
            ENDIF
            RETURN calculatedLoad
        ENDMETHOD
    ENDCLASS

    CLASS CacheServer EXTENDS DatabaseServer
        PRIVATE ATTRIBUTES:
            cacheHitRatio : Double // Range: 0.0 to 1.0

        CONSTRUCTOR(serial, uptime, specs, throughput, qps, hitRatio)
            CALL SUPER_CONSTRUCTOR(serial, uptime, specs, throughput, qps)
            SET this.cacheHitRatio TO hitRatio
        ENDCONSTRUCTOR

        METHOD calculateLoadPercentage()
            SET databaseLoad TO CALL SUPER_METHOD_calculateLoadPercentage()
            IF cacheHitRatio Greater Than 0.8 THEN
                RETURN databaseLoad * 0.5
            ELSE
                RETURN databaseLoad
            ENDIF
        ENDMETHOD
    ENDCLASS

    CLASS Rack
        PRIVATE ATTRIBUTES:
            rackID : Integer
            serversList : List of Server

        CONSTRUCTOR(id)
            SET this.rackID TO id
            SET this.serversList TO LIST []
        ENDCONSTRUCTOR

        METHOD installServer(serverObj)
            ADD serverObj TO this.serversList
        ENDMETHOD

        METHOD calculateAverageRackLoad()
            SET totalLoad TO 0.0
            SET totalServers TO SIZE OF this.serversList
            IF totalServers Equals 0 THEN
                RETURN 0.0
            ENDIF

            REPEAT FROM i = 0 TO totalServers - 1
                SET totalLoad TO totalLoad + CALL this.serversList[i].calculateLoadPercentage()
            ENDREPEAT

            RETURN totalLoad / totalServers
        ENDMETHOD
    ENDCLASS

    CREATE ServerSpecs FROM HardwareSpecification(64, 256, 12.0)
    CREATE MyCacheServer FROM CacheServer("SER-7015", 3600, ServerSpecs, 800.0, 1000.0, 0.9)
    
    CREATE DataRack FROM Rack(1)
    CALL DataRack.installServer(MyCacheServer)

    DISPLAY "Data Center Infrastructure Monitoring Station"
    DISPLAY "Optimized Cache Effective Load Level: " + CALL MyCacheServer.calculateLoadPercentage() + "%"
    DISPLAY "Aggregate Mean Hardware Rack Load: " + CALL DataRack.calculateAverageRackLoad() + "%"
